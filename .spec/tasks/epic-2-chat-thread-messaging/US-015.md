# Design ChatInput component with send button and empty validation

> Task ID: US-015
> Type: feature:design
> Priority: P1
> Appetite Fit: small (fits)
> Assignee: react-native-ui-implementer
> PRD Version: 3.1.0

## CRITICAL CONSTRAINTS

### MUST
- Create `components/chat/ChatInput.tsx` with a text input and send button
- Create `components/chat/ChatInput.stories.tsx` with Storybook stories
- Use semantic theme tokens for all colors, spacing, and typography
- Disable the send button when input is empty or whitespace-only
- Accept props: `value: string`, `onChangeText: (text: string) => void`, `onSend: () => void`, `disabled: boolean`
- Show a placeholder text: "Type a message..."
- Send button should be an icon button (send arrow icon from Lucide)
- Support keyboard "Enter" to submit (via `onSubmitEditing`)
- Include `testID` props: `chat-input-field`, `chat-input-send-button`

### NEVER
- Hardcode colors (use semantic tokens)
- Implement slash command detection in this task (that's Epic 3)
- Call the API directly (that's US-017 integration)
- Use a multi-line TextInput (single line for MVP)

### STRICTLY
- Stories must cover: Default, WithText, Disabled, SendButtonDisabled (empty)
- Play functions must verify the send button disabled state for empty input
- Use the existing `@/components/ui/input` component if available, otherwise use base TextInput with styling

## SPECIFICATION

**Objective:** Create the ChatInput component that provides a text input field with send button, validation for empty messages, and keyboard submit support — preparing for integration with chat-send API in US-017.

**Success looks like:** Storybook shows ChatInput rendering correctly. The send button is disabled when the input is empty. Stories demonstrate the various states. Play functions verify the disabled state logic. The component is ready for data wiring in US-017.

### Code Pattern

```typescript
// components/chat/ChatInput.tsx
import { View, TextInput, StyleSheet, Pressable } from 'react-native'
import { Send } from 'lucide-react-native'
import { useSemanticTheme } from '@/hooks/use-semantic-theme'

export interface ChatInputProps {
  value: string
  onChangeText: (text: string) => void
  onSend: () => void
  disabled?: boolean
  placeholder?: string
  testID?: string
}

export function ChatInput({
  value,
  onChangeText,
  onSend,
  disabled = false,
  placeholder = 'Type a message...',
  testID = 'chat-input',
}: ChatInputProps) {
  const { semantic } = useSemanticTheme()

  const isEmpty = value.trim().length === 0
  const canSend = !disabled && !isEmpty

  const handleSubmit = () => {
    if (canSend) {
      onSend()
    }
  }

  return (
    <View
      style={[
        styles.container,
        {
          backgroundColor: semantic.color.card.default,
          borderTopColor: semantic.color.border.default,
          borderTopWidth: 1,
          padding: semantic.space.sm,
        },
      ]}
      testID={testID}
    >
      <View
        style={[
          styles.inputWrapper,
          {
            backgroundColor: semantic.color.background.default,
            borderRadius: semantic.radius.lg,
            borderWidth: 1,
            borderColor: semantic.color.border.default,
            paddingHorizontal: semantic.space.md,
            paddingVertical: semantic.space.sm,
          },
        ]}
      >
        <TextInput
          style={[
            styles.input,
            {
              color: semantic.color.onSurface.default,
              flex: 1,
            },
          ]}
          value={value}
          onChangeText={onChangeText}
          onSubmitEditing={handleSubmit}
          placeholder={placeholder}
          placeholderTextColor={semantic.color.onSurface.muted}
          editable={!disabled}
          returnKeyType="send"
          testID="chat-input-field"
        />
        <Pressable
          onPress={handleSubmit}
          disabled={!canSend}
          testID="chat-input-send-button"
          style={({ pressed }) => [
            styles.sendButton,
            {
              backgroundColor: canSend
                ? pressed
                  ? semantic.color.primary.pressed
                  : semantic.color.primary.default
                : semantic.color.muted.default,
              borderRadius: semantic.radius.full,
              padding: semantic.space.sm,
            },
          ]}
        >
          <Send
            size={20}
            color={canSend ? semantic.color.onPrimary.default : semantic.color.onSurface.muted}
          />
        </Pressable>
      </View>
    </View>
  )
}

const styles = StyleSheet.create({
  container: {
    width: '100%',
  },
  inputWrapper: {
    flexDirection: 'row',
    alignItems: 'center',
  },
  input: {
    fontSize: 16,
    minHeight: 40,
  },
  sendButton: {
    marginLeft: 8,
  },
})
```

### Story Pattern

```typescript
// components/chat/ChatInput.stories.tsx
import type { Meta, StoryObj } from '@storybook/react'
import { useState } from 'react'
import { View } from 'react-native'
import { within, userEvent, expect } from '@storybook/test'
import { ChatInput } from './ChatInput'

const meta = {
  title: 'Chat/ChatInput',
  component: ChatInput,
  decorators: [
    (Story) => (
      <View style={{ width: '100%', padding: 16 }}>
        <Story />
      </View>
    ),
  ],
} satisfies Meta<typeof ChatInput>

export default meta
type Story = StoryObj<typeof meta>

export const Default: Story = {
  args: {
    value: '',
    onChangeText: () => {},
    onSend: () => console.log('Send pressed'),
    disabled: false,
  },
  play: async ({ canvasElement }) => {
    const canvas = within(canvasElement)
    const sendButton = canvas.getByTestId('chat-input-send-button')
    // Button should be disabled when empty
    await expect(sendButton).toBeDisabled()
  },
}

export const WithText: Story = {
  args: {
    value: 'Hello, can you help me?',
    onChangeText: () => {},
    onSend: () => console.log('Send pressed'),
    disabled: false,
  },
  play: async ({ canvasElement }) => {
    const canvas = within(canvasElement)
    const sendButton = canvas.getByTestId('chat-input-send-button')
    // Button should be enabled when there's text
    await expect(sendButton).not.toBeDisabled()
  },
}

export const Disabled: Story = {
  args: {
    value: 'Some text',
    onChangeText: () => {},
    onSend: () => {},
    disabled: true,
  },
}

export const WhitespaceOnly: Story = {
  args: {
    value: '   ',
    onChangeText: () => {},
    onSend: () => {},
    disabled: false,
  },
  play: async ({ canvasElement }) => {
    const canvas = within(canvasElement)
    const sendButton = canvas.getByTestId('chat-input-send-button')
    // Button should be disabled for whitespace-only
    await expect(sendButton).toBeDisabled()
  },
}

// Interactive example with state
const InteractiveChatInput = () => {
  const [value, setValue] = useState('')
  return (
    <ChatInput
      value={value}
      onChangeText={setValue}
      onSend={() => {
        console.log('Sent:', value)
        setValue('')
      }}
    />
  )
}

export const Interactive: Story = {
  render: () => <InteractiveChatInput />,
}
```
