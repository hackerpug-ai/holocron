# Wire ChatInput to chat-send API with optimistic updates

> Task ID: US-017
> Type: feature:integration
> Priority: P2
> Appetite Fit: medium (fits)
> Assignee: react-native-ui-implementer
> PRD Version: 3.1.0
> Stability: stable

## CRITICAL CONSTRAINTS

### MUST
- Create `useChatSend` custom hook for sending messages
- Call POST `/functions/v1/chat-send` endpoint
- Implement optimistic updates (show user message immediately)
- Handle API errors with rollback of optimistic update
- Integrate with `useChatHistory` to append new messages
- Disable input while send is in progress

### NEVER
- Never block UI while waiting for API response (use optimistic updates)
- Never lose user's message on error (provide retry mechanism)
- Never hardcode the Supabase URL -- use environment config

### STRICTLY
- User message appears instantly with pending state
- Agent response appears after API returns
- Failed messages show error state with retry option
- Clear input only after successful send initiation

## SPECIFICATION

**Objective:** Wire the `ChatInput` component to the `chat-send` Edge Function, implementing optimistic updates for instant feedback and proper error handling with retry capability.

**Success looks like:** User types a message, presses send, sees their message appear instantly, sees a typing indicator, then sees the agent response. On error, the message shows failed state with retry option.

### API Contract
```typescript
// POST /functions/v1/chat-send
// Request body:
interface ChatSendRequest {
  conversation_id: string
  content: string
  message_type?: 'text' | 'slash_command'
}

// Response:
interface ChatSendResponse {
  user_message_id: string
  agent_messages: Array<{
    id: string
    role: 'agent'
    content: string
    message_type: string
    created_at: string
  }>
}
```

### Hook Pattern
```typescript
// hooks/useChatSend.ts
import { useState, useCallback } from 'react'
import { supabase } from '@/lib/supabase'

interface UseChatSendOptions {
  conversationId: string
  onMessageSent?: (userMessage: ChatMessage, agentMessages: ChatMessage[]) => void
  onError?: (error: Error, pendingMessage: ChatMessage) => void
}

interface UseChatSendResult {
  send: (content: string) => Promise<void>
  isSending: boolean
  pendingMessage: ChatMessage | null
  error: Error | null
  retry: () => Promise<void>
  clearError: () => void
}

export function useChatSend({
  conversationId,
  onMessageSent,
  onError
}: UseChatSendOptions): UseChatSendResult {
  const [isSending, setIsSending] = useState(false)
  const [pendingMessage, setPendingMessage] = useState<ChatMessage | null>(null)
  const [error, setError] = useState<Error | null>(null)

  const send = useCallback(async (content: string) => {
    const trimmedContent = content.trim()
    if (!trimmedContent || isSending) return

    // Create optimistic message
    const optimisticMessage: ChatMessage = {
      id: `pending-${Date.now()}`,
      content: trimmedContent,
      role: 'user',
      created_at: new Date().toISOString(),
      isPending: true,
    }

    setPendingMessage(optimisticMessage)
    setIsSending(true)
    setError(null)

    try {
      const { data, error: sendError } = await supabase.functions.invoke('chat-send', {
        body: {
          conversation_id: conversationId,
          content: trimmedContent,
          message_type: 'text',
        },
      })

      if (sendError) throw sendError

      const response = data as ChatSendResponse

      // Update optimistic message with real ID
      const confirmedUserMessage: ChatMessage = {
        ...optimisticMessage,
        id: response.user_message_id,
        isPending: false,
      }

      onMessageSent?.(confirmedUserMessage, response.agent_messages)
      setPendingMessage(null)
    } catch (err) {
      const error = err instanceof Error ? err : new Error('Failed to send message')
      setError(error)
      onError?.(error, optimisticMessage)
    } finally {
      setIsSending(false)
    }
  }, [conversationId, isSending, onMessageSent, onError])

  const retry = useCallback(async () => {
    if (pendingMessage) {
      await send(pendingMessage.content)
    }
  }, [pendingMessage, send])

  const clearError = useCallback(() => {
    setError(null)
    setPendingMessage(null)
  }, [])

  return { send, isSending, pendingMessage, error, retry, clearError }
}
```

### Integration Pattern
```typescript
// app/(drawer)/conversation/[id].tsx
import { useChatHistory } from '@/hooks/useChatHistory'
import { useChatSend } from '@/hooks/useChatSend'
import { ChatThread } from '@/components/chat/ChatThread'
import { ChatInput } from '@/components/chat/ChatInput'

export default function ConversationScreen() {
  const { id } = useLocalSearchParams<{ id: string }>()

  const {
    messages,
    isLoading,
    hasMore,
    loadMore,
    appendMessages,
  } = useChatHistory({ conversationId: id })

  const {
    send,
    isSending,
    pendingMessage,
    error: sendError,
    retry,
  } = useChatSend({
    conversationId: id,
    onMessageSent: (userMsg, agentMsgs) => {
      appendMessages([userMsg, ...agentMsgs])
    },
  })

  // Include pending message in display
  const displayMessages = pendingMessage
    ? [...messages, pendingMessage]
    : messages

  return (
    <View className="flex-1">
      <ChatThread
        messages={displayMessages}
        isLoading={isLoading}
        hasMore={hasMore}
        onLoadMore={loadMore}
      />
      {sendError && (
        <ErrorBanner
          message="Failed to send message"
          onRetry={retry}
        />
      )}
      <ChatInput
        onSend={send}
        disabled={isSending}
      />
    </View>
  )
}
```

// DO NOT: Block until API responds
