# Wire ChatThread to chat-history API with pagination

> Task ID: US-016
> Type: feature:integration
> Priority: P2
> Appetite Fit: medium (fits)
> Assignee: react-native-ui-implementer
> PRD Version: 3.1.0
> Stability: stable

## CRITICAL CONSTRAINTS

### MUST
- Create `useChatHistory` custom hook for data fetching
- Call GET `/functions/v1/chat-history` endpoint with conversation_id
- Implement cursor-based pagination using `before` parameter
- Handle loading, error, and empty states
- Transform API response to ChatThread's expected format
- Support pull-to-refresh for latest messages

### NEVER
- Never use offset-based pagination
- Never fetch all messages at once (use pagination)
- Never duplicate message state between hook and component
- Never hardcode the Supabase URL -- use environment config

### STRICTLY
- Use `before` cursor from API response for next page
- Append older messages to existing array (not replace)
- Handle API errors gracefully with user-facing message
- Loading state must be visually distinct

## SPECIFICATION

**Objective:** Wire the `ChatThread` component to the `chat-history` Edge Function, implementing cursor-based pagination for infinite scroll loading of older messages.

**Success looks like:** The chat thread loads initial messages, supports scrolling up to load older messages, handles loading states, and maintains smooth performance with 100+ messages.

### API Contract
```typescript
// GET /functions/v1/chat-history?conversation_id=X&limit=20&before=cursor

// Response
interface ChatHistoryResponse {
  messages: Array<{
    id: string
    role: 'user' | 'agent' | 'system'
    content: string
    message_type: string
    card_data: unknown | null
    created_at: string
  }>
  has_more: boolean
  next_cursor: string | null
}
```

### Hook Pattern
```typescript
// hooks/useChatHistory.ts
import { useState, useCallback } from 'react'
import { supabase } from '@/lib/supabase'

interface UseChatHistoryOptions {
  conversationId: string
  limit?: number
}

interface UseChatHistoryResult {
  messages: ChatMessage[]
  isLoading: boolean
  isLoadingMore: boolean
  error: Error | null
  hasMore: boolean
  loadMore: () => Promise<void>
  refresh: () => Promise<void>
}

export function useChatHistory({ conversationId, limit = 20 }: UseChatHistoryOptions): UseChatHistoryResult {
  const [messages, setMessages] = useState<ChatMessage[]>([])
  const [cursor, setCursor] = useState<string | null>(null)
  const [hasMore, setHasMore] = useState(true)
  const [isLoading, setIsLoading] = useState(true)
  const [isLoadingMore, setIsLoadingMore] = useState(false)
  const [error, setError] = useState<Error | null>(null)

  const fetchMessages = useCallback(async (beforeCursor?: string, isRefresh = false) => {
    try {
      const params = new URLSearchParams({
        conversation_id: conversationId,
        limit: String(limit),
      })
      if (beforeCursor) {
        params.append('before', beforeCursor)
      }

      const { data, error: fetchError } = await supabase.functions.invoke('chat-history', {
        method: 'GET',
        body: null,
        // Note: query params passed differently for Edge Functions
      })

      if (fetchError) throw fetchError

      const response = data as ChatHistoryResponse

      if (isRefresh) {
        setMessages(response.messages.reverse()) // API returns newest first, we want oldest first
      } else {
        setMessages(prev => [...response.messages.reverse(), ...prev])
      }

      setHasMore(response.has_more)
      setCursor(response.next_cursor)
    } catch (err) {
      setError(err instanceof Error ? err : new Error('Failed to load messages'))
    }
  }, [conversationId, limit])

  const loadMore = useCallback(async () => {
    if (!hasMore || isLoadingMore) return
    setIsLoadingMore(true)
    await fetchMessages(cursor)
    setIsLoadingMore(false)
  }, [cursor, hasMore, isLoadingMore, fetchMessages])

  const refresh = useCallback(async () => {
    setIsLoading(true)
    setCursor(null)
    setHasMore(true)
    await fetchMessages(undefined, true)
    setIsLoading(false)
  }, [fetchMessages])

  // Initial load
  useEffect(() => {
    refresh()
  }, [conversationId])

  return { messages, isLoading, isLoadingMore, error, hasMore, loadMore, refresh }
}
```

### Integration Pattern
```typescript
// app/(drawer)/conversation/[id].tsx
import { useChatHistory } from '@/hooks/useChatHistory'
import { ChatThread } from '@/components/chat/ChatThread'

export default function ConversationScreen() {
  const { id } = useLocalSearchParams<{ id: string }>()

  const {
    messages,
    isLoading,
    isLoadingMore,
    error,
    hasMore,
    loadMore,
    refresh
  } = useChatHistory({ conversationId: id })

  if (error) {
    return <ErrorState message={error.message} onRetry={refresh} />
  }

  return (
    <ChatThread
      messages={messages}
      isLoading={isLoading || isLoadingMore}
      hasMore={hasMore}
      onLoadMore={loadMore}
    />
  )
}
```
