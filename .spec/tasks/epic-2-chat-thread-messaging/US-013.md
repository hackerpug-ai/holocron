> Task ID: US-013
> Type: feature:design

## CRITICAL CONSTRAINTS

### MUST
- Create `components/chat/MessageBubble.tsx` with three role variants: 'user', 'agent', 'system'
- Create `components/chat/MessageBubble.stories.tsx` with Storybook stories for all variants
- Use semantic theme tokens from `useSemanticTheme()` for all colors, spacing, and typography
- User messages: right-aligned, primary color background, white text
- Agent messages: left-aligned, card/surface background, foreground text
- System messages: center-aligned, muted background, muted text, smaller font
- Support markdown rendering in message content (bold, italic, code, links)
- Include a `testID` prop for testing: `message-bubble-{role}`
- Export component and types from `components/chat/index.ts` barrel

### NEVER
- Hardcode colors (use semantic tokens: `semantic.color.primary.default`, `semantic.color.card.default`, etc.)
- Hardcode spacing values (use `semantic.space.*`)
- Use react-native Text directly (use react-native-paper Text with variants)
- Include timestamp display in this component (that's a separate concern for US-018)

### STRICTLY
- Follow the DESIGN task pattern: component + story + play functions
- Stories must cover: Default (each role), AllVariants (grid), LongContent (edge case), WithMarkdown
- Play functions must verify render without errors for each variant
- Use `react-native-markdown-display` for markdown rendering (already in package.json)

## SPECIFICATION

**Objective:** Create the MessageBubble component that renders a single chat message with role-appropriate styling, enabling the ChatThread to display user, agent, and system messages distinctly.

**Success looks like:** Storybook shows MessageBubble rendering correctly for all three roles. User messages are right-aligned with primary styling, agent messages are left-aligned with card styling, system messages are centered with muted styling. Markdown content renders properly. Play functions pass in `vitest --project=storybook`.

### Code Pattern

```typescript
// components/chat/MessageBubble.tsx
import { View, StyleSheet } from 'react-native'
import { Text } from 'react-native-paper'
import Markdown from 'react-native-markdown-display'
import { useSemanticTheme } from '@/hooks/use-semantic-theme'

export type MessageRole = 'user' | 'agent' | 'system'

export interface MessageBubbleProps {
  role: MessageRole
  content: string
  testID?: string
}

export function MessageBubble({ role, content, testID }: MessageBubbleProps) {
  const { semantic } = useSemanticTheme()

  const containerStyle = [
    styles.container,
    role === 'user' && styles.userContainer,
    role === 'agent' && styles.agentContainer,
    role === 'system' && styles.systemContainer,
  ]

  const bubbleStyle = [
    styles.bubble,
    {
      backgroundColor: role === 'user'
        ? semantic.color.primary.default
        : role === 'agent'
        ? semantic.color.card.default
        : semantic.color.muted.default,
      borderRadius: semantic.radius.lg,
      padding: semantic.space.md,
      maxWidth: role === 'system' ? '80%' : '75%',
    },
  ]

  const textColor = role === 'user'
    ? semantic.color.onPrimary.default
    : role === 'system'
    ? semantic.color.onSurface.muted
    : semantic.color.onSurface.default

  return (
    <View
      style={containerStyle}
      testID={testID ?? `message-bubble-${role}`}
    >
      <View style={bubbleStyle}>
        <Markdown
          style={{
            body: { color: textColor, margin: 0 },
            strong: { fontWeight: '700' },
            em: { fontStyle: 'italic' },
            code_inline: {
              backgroundColor: semantic.color.muted.default,
              fontFamily: 'monospace',
              paddingHorizontal: semantic.space.xs,
            },
          }}
        >
          {content}
        </Markdown>
      </View>
    </View>
  )
}

const styles = StyleSheet.create({
  container: {
    marginVertical: 4,
    paddingHorizontal: 16,
  },
  userContainer: {
    alignItems: 'flex-end',
  },
  agentContainer: {
    alignItems: 'flex-start',
  },
  systemContainer: {
    alignItems: 'center',
  },
  bubble: {
    // Dynamic styles applied inline
  },
})
```

### Story Pattern

```typescript
// components/chat/MessageBubble.stories.tsx
import type { Meta, StoryObj } from '@storybook/react'
import { View } from 'react-native'
import { MessageBubble } from './MessageBubble'

const meta = {
  title: 'Chat/MessageBubble',
  component: MessageBubble,
} satisfies Meta<typeof MessageBubble>

export default meta
type Story = StoryObj<typeof meta>

export const UserMessage: Story = {
  args: {
    role: 'user',
    content: 'Hello, can you help me find research on AI agents?',
  },
}

export const AgentMessage: Story = {
  args: {
    role: 'agent',
    content: "I'd be happy to help! Let me search your knowledge base for relevant articles.",
  },
}

export const SystemMessage: Story = {
  args: {
    role: 'system',
    content: 'Research session started',
  },
}

export const WithMarkdown: Story = {
  args: {
    role: 'agent',
    content: 'Here are **key findings**:\n\n- *First* point\n- `code example`\n- [Link text](https://example.com)',
  },
}

export const LongContent: Story = {
  args: {
    role: 'agent',
    content: 'Lorem ipsum '.repeat(50),
  },
}

export const AllVariants: Story = {
  render: () => (
    <View style={{ gap: 8 }}>
      <MessageBubble role="user" content="User message example" />
      <MessageBubble role="agent" content="Agent message example" />
      <MessageBubble role="system" content="System message" />
    </View>
  ),
}
```
