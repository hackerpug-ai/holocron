# Add typing indicator and timestamp display to message bubbles

> Task ID: US-018
> Type: feature:integration

> Assignee: react-native-ui-implementer
> PRD Version: 3.1.0

## CRITICAL CONSTRAINTS

### MUST
- Add timestamp display below or alongside message bubbles
- Show relative timestamps for recent messages (e.g., "just now", "2 min ago")
- Show absolute timestamps for older messages (e.g., "Mar 1, 10:30 AM")
- Add a typing indicator that appears when the agent is "thinking"
- The typing indicator should be an animated visual (pulsing dots or similar)
- Typing indicator appears as a pseudo-message in the chat thread
- Set up Supabase Realtime subscription for new messages in the active conversation
- Auto-append new messages received via Realtime to the message list

### NEVER
- Block the UI during Realtime setup
- Show duplicate messages (dedup by message ID)
- Hardcode timestamp formats (use a date library like date-fns)
- Leave Realtime subscriptions open when conversation changes (cleanup required)

### STRICTLY
- Typing indicator appears immediately when user sends a message, disappears when agent response arrives
- Realtime subscription must filter by `conversation_id` to avoid receiving messages from other conversations
- Timestamps must update periodically for relative times (e.g., every minute)

## SPECIFICATION

**Objective:** Enhance the chat experience with visual polish — timestamps on messages, typing indicator during agent processing, and real-time message updates via Supabase Realtime subscriptions.

**Success looks like:** Messages display human-readable timestamps. When the user sends a message, a typing indicator appears immediately below their message. When the agent response arrives (via API or Realtime), the typing indicator disappears and the response appears. New messages from other sessions (if applicable) appear in real-time without page refresh.

### Code Pattern - TypingIndicator

```typescript
// components/chat/TypingIndicator.tsx
import { useEffect, useRef } from 'react'
import { View, StyleSheet, Animated } from 'react-native'
import { useSemanticTheme } from '@/hooks/use-semantic-theme'

export interface TypingIndicatorProps {
  testID?: string
}

export function TypingIndicator({ testID = 'typing-indicator' }: TypingIndicatorProps) {
  const { semantic } = useSemanticTheme()
  const dot1 = useRef(new Animated.Value(0)).current
  const dot2 = useRef(new Animated.Value(0)).current
  const dot3 = useRef(new Animated.Value(0)).current

  useEffect(() => {
    const animate = (dot: Animated.Value, delay: number) => {
      Animated.loop(
        Animated.sequence([
          Animated.delay(delay),
          Animated.timing(dot, {
            toValue: 1,
            duration: 300,
            useNativeDriver: true,
          }),
          Animated.timing(dot, {
            toValue: 0,
            duration: 300,
            useNativeDriver: true,
          }),
        ])
      ).start()
    }

    animate(dot1, 0)
    animate(dot2, 150)
    animate(dot3, 300)
  }, [])

  const dotStyle = (animValue: Animated.Value) => ({
    opacity: animValue.interpolate({
      inputRange: [0, 1],
      outputRange: [0.3, 1],
    }),
  })

  return (
    <View
      style={[
        styles.container,
        {
          backgroundColor: semantic.color.card.default,
          borderRadius: semantic.radius.lg,
          padding: semantic.space.md,
          marginLeft: semantic.space.lg,
        },
      ]}
      testID={testID}
    >
      <Animated.View
        style={[
          styles.dot,
          { backgroundColor: semantic.color.primary.default },
          dotStyle(dot1),
        ]}
      />
      <Animated.View
        style={[
          styles.dot,
          { backgroundColor: semantic.color.primary.default },
          dotStyle(dot2),
        ]}
      />
      <Animated.View
        style={[
          styles.dot,
          { backgroundColor: semantic.color.primary.default },
          dotStyle(dot3),
        ]}
      />
    </View>
  )
}

const styles = StyleSheet.create({
  container: {
    flexDirection: 'row',
    alignItems: 'center',
    alignSelf: 'flex-start',
    gap: 4,
  },
  dot: {
    width: 8,
    height: 8,
    borderRadius: 4,
  },
})
```

### Code Pattern - Timestamp

```typescript
// In MessageBubble.tsx, add timestamp display
import { formatDistanceToNow, format, isToday, isYesterday } from 'date-fns'

function formatTimestamp(dateString: string): string {
  const date = new Date(dateString)
  const now = new Date()
  const diffMinutes = Math.floor((now.getTime() - date.getTime()) / 60000)

  if (diffMinutes < 1) return 'just now'
  if (diffMinutes < 60) return `${diffMinutes} min ago`
  if (isToday(date)) return format(date, 'h:mm a')
  if (isYesterday(date)) return `Yesterday, ${format(date, 'h:mm a')}`
  return format(date, 'MMM d, h:mm a')
}

// Add to MessageBubble props
interface MessageBubbleProps {
  // ... existing props
  createdAt?: string
  showTimestamp?: boolean
}

// Render timestamp below bubble
{showTimestamp && createdAt && (
  <Text
    variant="labelSmall"
    style={{
      color: semantic.color.onSurface.muted,
      marginTop: semantic.space.xs,
      alignSelf: role === 'user' ? 'flex-end' : 'flex-start',
    }}
  >
    {formatTimestamp(createdAt)}
  </Text>
)}
```

### Code Pattern - Realtime Hook

```typescript
// hooks/use-chat-realtime.ts
import { useEffect, useCallback } from 'react'
import { supabase } from '@/lib/supabase'

interface UseChatRealtimeOptions {
  conversationId: string | null
  onNewMessage: (message: any) => void
}

export function useChatRealtime({ conversationId, onNewMessage }: UseChatRealtimeOptions) {
  useEffect(() => {
    if (!conversationId) return

    const channel = supabase
      .channel(`chat-${conversationId}`)
      .on(
        'postgres_changes',
        {
          event: 'INSERT',
          schema: 'public',
          table: 'chat_messages',
          filter: `conversation_id=eq.${conversationId}`,
        },
        (payload) => {
          onNewMessage(payload.new)
        }
      )
      .subscribe()

    return () => {
      supabase.removeChannel(channel)
    }
  }, [conversationId, onNewMessage])
}
```
