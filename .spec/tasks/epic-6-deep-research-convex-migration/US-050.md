## DESIGN

### Pattern (Validation Script)

```typescript
// scripts/validate-migration.ts
import { createClient } from "@supabase/supabase-js";
import { ConvexHttpClient } from "convex/browser";

interface ValidationResult {
  table: string;
  check: string;
  status: "PASS" | "FAIL";
  expected: number | string;
  actual: number | string;
  details?: string;
}

const results: ValidationResult[] = [];

// Check 1: Row counts
async function validateRowCounts() {
  const tables = [
    { supabase: "conversations", convex: "conversations" },
    { supabase: "chat_messages", convex: "chatMessages" },
    { supabase: "documents", convex: "documents" },
    { supabase: "research_sessions", convex: "researchSessions" },
    { supabase: "research_iterations", convex: "researchIterations" },
    { supabase: "deep_research_sessions", convex: "deepResearchSessions" },
    { supabase: "deep_research_iterations", convex: "deepResearchIterations" },
    { supabase: "citations", convex: "citations" },
    { supabase: "long_running_tasks", convex: "tasks" },
  ];

  for (const { supabase: sTable, convex: cTable } of tables) {
    const { count: sCount } = await supabase
      .from(sTable)
      .select("*", { count: "exact", head: true });
    const cCount = await convex.query(api[cTable].count);

    results.push({
      table: cTable,
      check: "row_count",
      status: sCount === cCount ? "PASS" : "FAIL",
      expected: sCount!,
      actual: cCount,
    });
  }
}

// Check 2: FK relationships
async function validateForeignKeys() {
  // chatMessages → conversations
  const messagesWithoutConv = await convex.query(api.chatMessages.getOrphans);
  results.push({
    table: "chatMessages",
    check: "fk_conversations",
    status: messagesWithoutConv.length === 0 ? "PASS" : "FAIL",
    expected: 0,
    actual: messagesWithoutConv.length,
    details: messagesWithoutConv.length > 0 ? JSON.stringify(messagesWithoutConv.slice(0, 5)) : undefined,
  });

  // Similar for all other FK relationships...
}

// Check 3: Embedding dimensions
async function validateEmbeddingDimensions() {
  const documents = await convex.query(api.documents.listWithEmbeddings);

  const invalidEmbeddings = documents.filter(
    (d) => d.embedding && d.embedding.length !== 1536
  );

  results.push({
    table: "documents",
    check: "embedding_dimensions",
    status: invalidEmbeddings.length === 0 ? "PASS" : "FAIL",
    expected: 1536,
    actual: invalidEmbeddings.length === 0 ? 1536 : `varies (${invalidEmbeddings.length} invalid)`,
    details: invalidEmbeddings.length > 0 ? `Invalid: ${invalidEmbeddings.map(d => d._id).join(", ")}` : undefined,
  });
}

// Check 4: Chat message ordering
async function validateChatOrdering() {
  const conversations = await convex.query(api.conversations.list);

  for (const conv of conversations) {
    const messages = await convex.query(api.chatMessages.listByConversation, {
      conversationId: conv._id,
    });

    // Check ascending order by createdAt
    for (let i = 1; i < messages.length; i++) {
      if (messages[i].createdAt < messages[i - 1].createdAt) {
        results.push({
          table: "chatMessages",
          check: "ordering",
          status: "FAIL",
          expected: "ascending",
          actual: "out_of_order",
          details: `Conversation ${conv._id}: messages out of order`,
        });
        break;
      }
    }
  }

  // If no failures recorded, add pass
  if (!results.some(r => r.check === "ordering" && r.status === "FAIL")) {
    results.push({
      table: "chatMessages",
      check: "ordering",
      status: "PASS",
      expected: "ascending",
      actual: "ascending",
    });
  }
}

// Generate report
async function generateReport() {
  const passCount = results.filter((r) => r.status === "PASS").length;
  const failCount = results.filter((r) => r.status === "FAIL").length;
  const passRate = (passCount / results.length) * 100;

  const report = {
    timestamp: new Date().toISOString(),
    summary: {
      total: results.length,
      passed: passCount,
      failed: failCount,
      passRate: `${passRate.toFixed(1)}%`,
    },
    results,
  };

  await Bun.write(".tmp/validation-report.json", JSON.stringify(report, null, 2));

  console.log("\n=== VALIDATION REPORT ===");
  console.log(`Total checks: ${results.length}`);
  console.log(`Passed: ${passCount}`);
  console.log(`Failed: ${failCount}`);
  console.log(`Pass rate: ${passRate.toFixed(1)}%`);

  if (failCount > 0) {
    console.log("\n=== FAILED CHECKS ===");
    results
      .filter((r) => r.status === "FAIL")
      .forEach((r) => {
        console.log(`❌ ${r.table}.${r.check}: expected ${r.expected}, got ${r.actual}`);
        if (r.details) console.log(`   Details: ${r.details}`);
      });
  }
}
```
