## DESIGN
### Pattern (Iteration Timeline)
```typescript
// components/research/IterationTimeline.tsx
import { View, StyleSheet, ScrollView, Pressable } from 'react-native';
import { Text, Card, Button } from 'react-native-paper';
import { useSemanticTheme } from '@/hooks/use-semantic-theme';
import { ScoreProgressIndicator } from './ScoreProgressIndicator';

interface IterationTimelineProps {
  iterations: Array<{
    iterationNumber: number;
    coverageScore: number;
    findings: string[];
    timestamp: number;
  }>;
  totalCost: {
    actual: string;      // "$0.32"
    baseline: string;     // "$0.90" (GPT-5-only)
    savings: string;       // "64%"
  };
  onDownloadReport: () => void;
}

export function IterationTimeline({
  iterations,
  totalCost,
  onDownloadReport,
}: IterationTimelineProps) {
  const { semantic } = useSemanticTheme();
  const [expandedIteration, setExpandedIteration] = useState<number | null>(null);

  return (
    <View style={[
      styles.container,
      {
        backgroundColor: semantic.color.surface.default,
        borderRadius: semantic.radius.lg,
        padding: semantic.space.lg,
      }
    ]}>
      {/* Header with Cost Summary */}
      <View style={styles.header}>
        <Text variant="titleMedium" style={{ color: semantic.color.onSurface.default }}>
          Research Timeline
        </Text>
        <View style={styles.costSummary}>
          <Text variant="labelSmall" style={{ color: semantic.color.onSurface.muted }}>
            Total Cost
          </Text>
          <View style={styles.costRow}>
            <Text variant="titleSmall" style={{ color: semantic.color.success.default }}>
              {totalCost.actual}
            </Text>
            <Text variant="labelSmall" style={{ color: semantic.color.onSurface.muted }}>
              vs {totalCost.baseline} (GPT-5 only)
            </Text>
          </View>
          <View style={[
            styles.savingsBadge,
            { backgroundColor: semantic.color.success.default }
          ]}>
            <Text variant="labelSmall" style={{ color: semantic.color.onPrimary.default }}>
              {totalCost.savings} saved
            </Text>
          </View>
        </View>
      </View>

      {/* Score Progress Indicator */}
      <ScoreProgressIndicator
        scores={iterations.map(i => i.coverageScore)}
        targetScore={4}
      />

      {/* Iteration Cards */}
      <ScrollView style={styles.timeline}>
        {iterations.map((iteration, index) => (
          <Pressable
            key={iteration.iterationNumber}
            onPress={() => setExpandedIteration(
              expandedIteration === iteration.iterationNumber ? null : iteration.iterationNumber
            )}
          >
            <IterationCard
              iteration={iteration}
              isExpanded={expandedIteration === iteration.iterationNumber}
              isLast={index === iterations.length - 1}
            />
          </Pressable>
        ))}
      </ScrollView>

      {/* Download Button */}
      <Button
        mode="contained"
        onPress={onDownloadReport}
        style={{ marginTop: semantic.space.md }}
      >
        Download Full Report
      </Button>
    </View>
  );
}

// components/research/ScoreProgressIndicator.tsx
export function ScoreProgressIndicator({
  scores,
  targetScore,
}: {
  scores: number[];
  targetScore: number;
}) {
  const { semantic } = useSemanticTheme();

  return (
    <View style={styles.progressContainer}>
      <Text variant="labelSmall" style={{ color: semantic.color.onSurface.muted }}>
        Coverage Progress (target: {targetScore}+)
      </Text>
      <View style={styles.scoreRow}>
        {scores.map((score, i) => (
          <View key={i} style={styles.scorePoint}>
            <View style={[
              styles.scoreDot,
              {
                backgroundColor: score >= targetScore
                  ? semantic.color.success.default
                  : score >= 3
                  ? semantic.color.warning.default
                  : semantic.color.error.default,
              }
            ]}>
              <Text variant="labelSmall" style={{ color: semantic.color.onPrimary.default }}>
                {score}
              </Text>
            </View>
            {i < scores.length - 1 && (
              <View style={[
                styles.connector,
                {
                  backgroundColor: scores[i + 1] > scores[i]
                    ? semantic.color.success.default
                    : semantic.color.border.default,
                }
              ]} />
            )}
          </View>
        ))}
      </View>
    </View>
  );
}
```
