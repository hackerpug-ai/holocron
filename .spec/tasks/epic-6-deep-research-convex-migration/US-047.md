## DESIGN

### Pattern (Convex Agent + Workflow)

```typescript
// convex/research/agents.ts
import { Agent } from "@convex-dev/agent";
import { openai } from "@ai-sdk/openai";

// Lead Agent - orchestrates research (GPT-5)
export const leadResearcher = new Agent(components.agent, {
  name: "Lead Researcher",
  chat: openai.chat("gpt-5"),
  textEmbedding: openai.embedding("text-embedding-3-small"),
  instructions: `You are a research coordinator. Analyze queries,
    decompose into subtasks, delegate to subagents, and synthesize findings.`,
  tools: { planResearch, synthesizeFindings, assessCoverage },
});

// Subagent - parallel workers (GPT-5-mini for cost)
export const webSearcher = new Agent(components.agent, {
  name: "Web Searcher",
  chat: openai.chat("gpt-5-mini"),
  instructions: `You are a focused research worker. Search for specific
    information on your assigned sub-topic. Cite all sources.`,
  tools: { searchWeb, readUrl, evaluateSource },
});

// Reviewer Agent (GPT-5)
export const reviewerAgent = new Agent(components.agent, {
  name: "Reviewer",
  chat: openai.chat("gpt-5"),
  instructions: `Assess coverage (1-5), identify gaps, decide iterate or complete.`,
});
```

```typescript
// convex/research/deepResearch.ts
import { WorkflowManager } from "@convex-dev/workflow";

export const deepResearch = workflow.define({
  args: { query: v.string(), userId: v.string(), maxIterations: v.number() },
  handler: async (step, { query, userId, maxIterations }) => {
    // Step 1: Lead agent plans research (GPT-5)
    const plan = await step.run("plan", async () => {
      const { thread } = await leadResearcher.createThread(ctx, { userId });
      return thread.generateObject({
        prompt: `Plan comprehensive research for: ${query}`,
        schema: ResearchPlanSchema,
      });
    });

    // Step 2: Spawn parallel subagents (GPT-5-mini)
    const findings = await Promise.all(
      plan.subtasks.map((task, i) =>
        step.run(`subagent-${i}`, async () => {
          const { thread } = await webSearcher.createThread(ctx, { userId });
          return thread.generateText({
            prompt: task.objective,
            tools: { searchWeb, readUrl },
            maxToolRoundtrips: 5, // Stay within 10-min limit
          });
        })
      )
    );

    // Step 3: Synthesize (GPT-5)
    const report = await step.run("synthesize", async () => {
      return leadResearcher.generateText({
        prompt: `Synthesize findings:\n${findings.join("\n---\n")}`,
      });
    });

    // Step 4: Review coverage
    const review = await step.run("review", async () => {
      return reviewerAgent.generateObject({
        prompt: `Assess coverage:\n${report}`,
        schema: ReviewSchema,
      });
    });

    // Iterate if coverage < 4
    if (review.score < 4 && step.iteration < maxIterations) {
      return step.continue({ refinedQueries: review.gaps });
    }

    return { report, review, iterations: step.iteration };
  },
});
```

### Anti-pattern (DO NOT)

```typescript
// BAD: Using GPT-5 for subagents (expensive)
chat: openai.chat("gpt-5") // Subagents should use gpt-5-mini

// BAD: Unlimited tool roundtrips (exceeds 10-min action limit)
maxToolRoundtrips: 20 // WRONG - will timeout

// BAD: Sequential subagent execution (slow)
for (const task of tasks) {
  await runSubagent(task); // WRONG - should be parallel
}
```
