// hooks/useLongRunningTask.ts
import { useEffect, useState, useCallback, useRef } from 'react';
import { supabase } from '@/lib/supabase';
import { taskRealtimeRegistry } from './taskRealtimeRegistry';

export function useLongRunningTask(options: {
  taskId?: string;
  conversationId?: string;
  onUpdate?: (task: Task) => void;
  onComplete?: (result: any) => void;
  onError?: (error: string) => void;
}) {
  const [task, setTask] = useState<Task | null>(null);
  const [status, setStatus] = useState<TaskStatus>('pending');
  const [progress, setProgress] = useState<Progress | null>(null);
  const channelRef = useRef<RealtimeChannel | null>(null);

  useEffect(() => {
    if (!options.taskId && !options.conversationId) return;

    const channel = supabase.channel(`task-${options.taskId || options.conversationId}`);
    channelRef.current = channel;

    channel
      .on('broadcast', { event: 'progress' }, (payload) => {
        setProgress(payload.payload);
        options.onUpdate?.(payload.payload);
      })
      .on('broadcast', { event: 'complete' }, (payload) => {
        setStatus('completed');
        options.onComplete?.(payload.payload);
      })
      .on('broadcast', { event: 'error' }, (payload) => {
        setStatus('error');
        options.onError?.(payload.payload.message);
      })
      .subscribe();

    taskRealtimeRegistry.register(channel);

    return () => {
      taskRealtimeRegistry.unregister(channel);
      channel.unsubscribe();
    };
  }, [options.taskId, options.conversationId]);

  return { task, status, progress };
}

// components/research/ResearchProgress.tsx
import { useQuery } from 'convex/react';
import { api } from '@/convex/_generated/api';

export function ResearchProgress({ taskId }: { taskId: string }) {
  // Single query - Convex auto-updates when entity changes
  const task = useQuery(api.tasks.get, { id: taskId });

  if (task === undefined) {
    return <ProgressSkeleton />;
  }

  // Component re-renders automatically as status/progress/result changes
  if (task.status === 'running') {
    return (
      <View>
        <ProgressBar message={task.progressMessage} />
        <Text>Step {task.currentStep} of {task.totalSteps}</Text>
      </View>
    );
  }

  if (task.status === 'completed') {
    return <ResearchResults data={task.result} />;
  }

  if (task.status === 'error') {
    return <ErrorDisplay message={task.errorMessage} />;
  }

  return <Text>Waiting...</Text>;
}
