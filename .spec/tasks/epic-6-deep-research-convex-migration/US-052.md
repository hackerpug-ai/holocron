## CRITICAL CONSTRAINTS

MUST: Use Convex Action for AI operations (side effects)
MUST: Support streaming via Convex HTTP endpoints
MUST: Persist user message before AI response
MUST: Handle slash command routing
MUST: Support message types: text, slash_command, result_card, progress, error

STRICTLY: Store message ordering via createdAt timestamp

## DESIGN

### Pattern (Convex Chat)

```typescript
// convex/chatMessages/queries.ts
import { query } from "./_generated/server";
import { v } from "convex/values";

export const listByConversation = query({
  args: {
    conversationId: v.id("conversations"),
    limit: v.optional(v.number()),
  },
  handler: async (ctx, { conversationId, limit = 50 }) => {
    return await ctx.db
      .query("chatMessages")
      .withIndex("by_conversation", (q) =>
        q.eq("conversationId", conversationId)
      )
      .order("desc")
      .take(limit);
  },
});

// convex/chat/send.ts
import { action } from "./_generated/server";
import { v } from "convex/values";
import { api } from "./_generated/api";

export const send = action({
  args: {
    conversationId: v.id("conversations"),
    content: v.string(),
    messageType: v.optional(v.union(
      v.literal("text"),
      v.literal("slash_command")
    )),
  },
  handler: async (ctx, { conversationId, content, messageType = "text" }) => {
    const now = Date.now();

    // 1. Persist user message
    const userMessageId = await ctx.runMutation(api.chatMessages.create, {
      conversationId,
      role: "user",
      content,
      messageType,
      createdAt: now,
    });

    // 2. Update conversation updatedAt
    await ctx.runMutation(api.conversations.touch, { id: conversationId });

    // 3. Route message
    let agentResponse: string;
    let responseMessageType: "text" | "result_card" = "text";
    let cardData: any = undefined;

    if (content.startsWith("/")) {
      // Slash command routing
      const result = await routeSlashCommand(ctx, conversationId, content);
      agentResponse = result.message;
      responseMessageType = result.messageType;
      cardData = result.cardData;
    } else {
      // Natural language chat
      agentResponse = await generateAIResponse(content);
    }

    // 4. Persist agent response
    const agentMessageId = await ctx.runMutation(api.chatMessages.create, {
      conversationId,
      role: "agent",
      content: agentResponse,
      messageType: responseMessageType,
      cardData,
      createdAt: Date.now(),
    });

    return {
      userMessageId,
      agentMessageId,
      agentResponse,
    };
  },
});

// convex/http.ts - Streaming endpoint
import { httpRouter } from "convex/server";
import { api } from "./_generated/api";

const http = httpRouter();

http.route({
  path: "/chat/stream",
  method: "POST",
  handler: async (ctx, request) => {
    const { conversationId, content } = await request.json();

    // Use streaming AI SDK integration
    const stream = await streamAIResponse(content);

    return new Response(stream, {
      headers: {
        "Content-Type": "text/event-stream",
        "Cache-Control": "no-cache",
      },
    });
  },
});

export default http;
```

### Anti-pattern (DO NOT)

```typescript
// BAD: Blocking on AI response
await generateAIResponse(content); // UI frozen!

// BAD: Not persisting user message first
// User message must be persisted before AI call

// BAD: Not handling slash commands
// All / prefixed messages need routing
```
