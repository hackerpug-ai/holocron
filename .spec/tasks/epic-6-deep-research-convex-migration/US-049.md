## DESIGN

### Pattern (Migration Pipeline)

```typescript
// scripts/migrate-from-supabase.ts
import { createClient } from "@supabase/supabase-js";
import { ConvexHttpClient } from "convex/browser";

const supabase = createClient(
  process.env.SUPABASE_URL!,
  process.env.SUPABASE_SERVICE_ROLE_KEY!
);

const convex = new ConvexHttpClient(process.env.CONVEX_URL!);

// Step 1: Export all tables
async function exportTables() {
  const tables = [
    "conversations",
    "chat_messages",
    "documents",
    "research_sessions",
    "research_iterations",
    "deep_research_sessions",
    "deep_research_iterations",
    "citations",
    "long_running_tasks",
  ];

  for (const table of tables) {
    const { data, error } = await supabase.from(table).select("*");
    if (error) throw error;

    await Bun.write(
      `.tmp/export/${table}.json`,
      JSON.stringify(data, null, 2)
    );
    console.log(`Exported ${table}: ${data.length} rows`);
  }
}

// Step 2: Build ID mapping (UUID → Convex ID)
async function buildIdMapping() {
  const mapping: Record<string, Record<string, string>> = {};

  // Import tables in dependency order and build mapping
  const order = [
    "conversations",
    "documents",
    "research_sessions",
    "chat_messages",
    "research_iterations",
    "deep_research_sessions",
    "deep_research_iterations",
    "citations",
    "tasks",
  ];

  for (const table of order) {
    mapping[table] = {};
    const data = JSON.parse(await Bun.file(`.tmp/export/${table}.json`).text());

    for (const row of data) {
      const convexId = await convex.mutation(api[table].create, {
        ...transformRow(row, mapping),
      });
      mapping[table][row.id] = convexId;
    }
  }

  await Bun.write(".tmp/mapping.json", JSON.stringify(mapping, null, 2));
}

// Step 3: Transform row (convert UUIDs to Convex IDs)
function transformRow(row: any, mapping: any): any {
  const transformed = { ...row };

  // Replace FK UUIDs with Convex IDs
  if (row.conversation_id && mapping.conversations[row.conversation_id]) {
    transformed.conversationId = mapping.conversations[row.conversation_id];
    delete transformed.conversation_id;
  }

  // Similar for other FKs...

  // Preserve embeddings
  if (row.embedding) {
    transformed.embedding = row.embedding; // Keep as-is
  }

  return transformed;
}

// Step 4: Verify migration
async function verifyMigration() {
  const tables = [
    { supabase: "conversations", convex: "conversations" },
    { supabase: "chat_messages", convex: "chatMessages" },
    // ... etc
  ];

  for (const { supabase: sTable, convex: cTable } of tables) {
    const { count: sCount } = await supabase
      .from(sTable)
      .select("*", { count: "exact", head: true });
    const cCount = await convex.query(api[cTable].count);

    if (sCount !== cCount) {
      throw new Error(`Count mismatch: ${sTable}=${sCount}, ${cTable}=${cCount}`);
    }
    console.log(`✓ ${sTable}: ${sCount} rows`);
  }
}
```

### Anti-pattern (DO NOT)

```typescript
// BAD: Modifying source data
await supabase.from("documents").update({ migrated: true }); // NEVER

// BAD: Not preserving embeddings
embedding: row.embedding?.slice(0, 512) // WRONG - must preserve 1536

// BAD: Non-idempotent migration
// Always check if data exists before inserting
```
