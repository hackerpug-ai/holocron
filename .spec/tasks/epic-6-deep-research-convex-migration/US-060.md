## DESIGN
### Pattern (Direct Entity Watching)
```typescript
// convex/researchSessions/queries.ts
import { query } from "../_generated/server";
import { v } from "convex/values";

export const get = query({
  args: { id: v.id("researchSessions") },
  handler: async (ctx, { id }) => {
    const session = await ctx.db.get(id);
    if (!session) {
      throw new Error(`Research session ${id} not found`);
    }

    // Also fetch iterations if deep research
    const iterations = session.researchType === "deep"
      ? await ctx.db
          .query("researchIterations")
          .withIndex("by_session", (q) => q.eq("sessionId", id))
          .collect()
      : undefined;

    return {
      ...session,
      iterations,
    };
  },
});

// components/research/ResearchProgress.tsx
import { useQuery } from 'convex/react';
import { api } from '@/convex/_generated/api';

export function ResearchProgress({ sessionId }: { sessionId: string }) {
  // Single query - Convex auto-updates when entity changes!
  const session = useQuery(api.researchSessions.get, { id: sessionId });

  // Handle loading state
  if (session === undefined) {
    return <ProgressSkeleton />;
  }

  // Handle different states
  switch (session.status) {
    case 'pending':
      return <WaitingIndicator message="Starting research..." />;

    case 'searching':
      return (
        <View>
          <ProgressBar indeterminate />
          <Text>Searching for information...</Text>
        </View>
      );

    case 'analyzing':
      return (
        <View>
          <ProgressBar indeterminate />
          <Text>Analyzing sources...</Text>
        </View>
      );

    case 'synthesizing':
      return (
        <View>
          <ProgressBar indeterminate />
          <Text>Synthesizing findings...</Text>
        </View>
      );

    case 'completed':
      return <ResearchResults session={session} />;

    case 'failed':
      return <ErrorDisplay message={session.errorText || 'Research failed'} />;

    case 'cancelled':
      return <CancelledNotice />;

    default:
      return <Text>Unknown status: {session.status}</Text>;
  }
}

// For deep research with iterations
export function DeepResearchProgress({ sessionId }: { sessionId: string }) {
  const session = useQuery(api.researchSessions.get, { id: sessionId });

  if (session === undefined) {
    return <ProgressSkeleton />;
  }

  return (
    <View>
      {/* Current iteration progress */}
      <IterationProgress
        current={session.currentIteration || 0}
        max={session.maxIterations || 5}
        status={session.status}
      />

      {/* Iteration timeline */}
      {session.iterations && session.iterations.length > 0 && (
        <IterationTimeline iterations={session.iterations} />
      )}

      {/* Coverage score */}
      {session.coverageScore && (
        <CoverageIndicator score={session.coverageScore} />
      )}

      {/* Final results */}
      {session.status === 'completed' && (
        <ResearchResults session={session} />
      )}
    </View>
  );
}
```
