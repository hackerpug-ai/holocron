## DESIGN
### Pattern (Full Workflow)
```typescript
// convex/research/deepResearch.ts
import { action } from "../_generated/server";
import { v } from "convex/values";
import { workflow } from "@convex-dev/workflow";
import { api } from "../_generated/api";
import { leadResearcher, webSearcher, reviewerAgent } from "./agents";

export const start = action({
  args: {
    query: v.string(),
    conversationId: v.id("conversations"),
    maxIterations: v.optional(v.number()),
  },
  handler: async (ctx, { query, conversationId, maxIterations = 5 }) => {
    // Create session record
    const sessionId = await ctx.runMutation(api.deepResearchSessions.create, {
      conversationId,
      topic: query,
      maxIterations,
      status: "pending",
    });

    // Start workflow
    await ctx.scheduler.runAfter(0, api.deepResearch.iterate, {
      sessionId,
      query,
      iteration: 1,
      maxIterations,
    });

    return { sessionId, status: "started" };
  },
});

export const iterate = action({
  args: {
    sessionId: v.id("deepResearchSessions"),
    query: v.string(),
    iteration: v.number(),
    maxIterations: v.number(),
    previousFindings: v.optional(v.array(v.string())),
  },
  handler: async (ctx, args) => {
    const { sessionId, query, iteration, maxIterations, previousFindings = [] } = args;

    // Step 1: Lead Agent plans research (GPT-5)
    const plan = await leadResearcher.generateObject({
      prompt: `Plan comprehensive research for: ${query}
        Previous findings: ${previousFindings.join("\n")}
        Generate 3-5 focused subtasks.`,
      schema: ResearchPlanSchema,
    });

    // Step 2: Spawn parallel subagents (GPT-5-mini)
    const findings = await Promise.all(
      plan.subtasks.map((task, i) =>
        webSearcher.generateText({
          prompt: task.objective,
          tools: { searchWeb, readUrl, evaluateSource },
          maxToolRoundtrips: 5,
        })
      )
    );

    // Step 3: Synthesize (GPT-5)
    const report = await leadResearcher.generateText({
      prompt: `Synthesize findings:\n${findings.join("\n---\n")}`,
    });

    // Step 4: Review coverage (GPT-5)
    const review = await reviewerAgent.generateObject({
      prompt: `Assess coverage:\n${report}`,
      schema: ReviewSchema,
    });

    // Store iteration
    await ctx.runMutation(api.deepResearchIterations.create, {
      sessionId,
      iterationNumber: iteration,
      coverageScore: review.score,
      feedback: review.feedback,
      findings: report,
      refinedQueries: review.gaps,
      status: "completed",
    });

    // Iterate or complete
    if (review.score < 4 && iteration < maxIterations) {
      await ctx.scheduler.runAfter(1, api.deepResearch.iterate, {
        sessionId,
        query,
        iteration: iteration + 1,
        maxIterations,
        previousFindings: [...previousFindings, report],
      });
    } else {
      await ctx.runMutation(api.deepResearchSessions.complete, {
        sessionId,
        status: "completed",
      });
    }
  },
});
```

### Anti-pattern (DO NOT)
```typescript
// BAD: Using GPT-5 for subagents (expensive)
const subagent = new Agent(components.agent, {
  chat: openai.chat("gpt-5"), // WRONG - use gpt-5-mini
});

// BAD: Unlimited tool roundtrips (exceeds 10-min limit)
maxToolRoundtrips: 20 // WRONG - will timeout

// BAD: Sequential subagent execution (slow)
for (const task of tasks) {
  await runSubagent(task); // WRONG - should be parallel
}
```
