## DESIGN

### Pattern (from Convex vector docs)

```typescript
// convex/schema.ts
import { defineSchema, defineTable } from "convex/server";
import { v } from "convex/values";

export default defineSchema({
  documents: defineTable({
    title: v.string(),
    content: v.string(),
    category: v.string(),
    filePath: v.optional(v.string()),
    fileType: v.optional(v.string()),
    status: v.optional(v.string()),
    date: v.optional(v.string()),
    time: v.optional(v.string()),
    researchType: v.optional(v.string()),
    iterations: v.optional(v.number()),
    embedding: v.optional(v.array(v.float64())), // 1536 dimensions
    createdAt: v.number(),
  }).searchIndex("by_category", {
    searchField: "category",
  }),
});
```

### Anti-pattern (DO NOT)

```typescript
// BAD: Using v.string() for embeddings
embedding: v.string() // WRONG - must be array of float64

// BAD: Not validating dimensions
// Always check embedding.length === 1536
```
