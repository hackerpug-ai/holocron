## DESIGN
### Pattern (Direct Convex Usage)
```typescript
// BEFORE (Supabase + React Query - DELETE THIS FILE)
// hooks/useConversations.ts
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { supabase } from '@/lib/supabase';

export function useConversations() {
  const queryClient = useQueryClient();

  const { data: conversations } = useQuery({
    queryKey: ['conversations'],
    queryFn: async () => {
      const { data, error } = await supabase
        .from('conversations')
        .select('*')
        .order('updated_at', { ascending: false });
      if (error) throw error;
      return data;
    },
  });

  const create = useMutation({
    mutationFn: async (title?: string) => {
      const { data, error } = await supabase
        .from('conversations')
        .insert({ title: title || 'New Chat' })
        .select()
        .single();
      if (error) throw error;
      return data;
    },
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['conversations'] });
    },
  });

  return { conversations, create };
}

// AFTER (Convex - USE DIRECTLY IN COMPONENTS)
// components/conversation/ConversationList.tsx
import { useQuery, useMutation } from 'convex/react';
import { api } from '@/convex/_generated/api';

export function ConversationList() {
  const conversations = useQuery(api.conversations.list, { limit: 50 });
  const createConversation = useMutation(api.conversations.create);

  const handleCreate = async () => {
    await createConversation({ title: 'New Chat' });
    // No need to invalidate - Convex updates automatically!
  };

  if (conversations === undefined) {
    return <ConversationListSkeleton />;
  }

  return (
    <View>
      {conversations.map((conv) => (
        <ConversationItem key={conv._id} conversation={conv} />
      ))}
      <Button onPress={handleCreate}>New Chat</Button>
    </View>
  );
}
```

### Anti-pattern (DO NOT)
```typescript
// BAD: Creating a wrapper hook around Convex
export function useConversations() {
  const conversations = useQuery(api.conversations.list);
  return { conversations }; // Unnecessary abstraction
}

// BAD: Manual invalidation
const create = useMutation(api.conversations.create);
await create({ title: 'Test' });
// No invalidation needed - Convex handles this!

// BAD: Keeping Supabase imports
import { supabase } from '@/lib/supabase'; // DELETE THIS
```
