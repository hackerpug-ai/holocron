## DESIGN

### Pattern (Convex Search)

```typescript
// convex/documents/search.ts
import { query } from "../_generated/server";
import { v } from "convex/values";
import { api } from "../_generated/api";

// Vector search using Convex vector index
export const vectorSearch = query({
  args: {
    embedding: v.array(v.float64()),
    limit: v.optional(v.number()),
    category: v.optional(v.string()),
  },
  handler: async (ctx, { embedding, limit = 10, category }) => {
    let query = ctx.db
      .query("documents")
      .withIndex("by_embedding", (q) =>
        q.vector("embedding", embedding)
      );

    if (category) {
      // Additional filtering (Convex supports this)
      query = query.filter((q) => q.eq(q.field("category"), category));
    }

    const results = await query.take(limit);

    // Calculate similarity scores
    return results.map(doc => ({
      ...doc,
      score: cosineSimilarity(embedding, doc.embedding || []),
    }));
  },
});

// Full-text search using Convex searchIndex
export const fullTextSearch = query({
  args: {
    query: v.string(),
    limit: v.optional(v.number()),
    category: v.optional(v.string()),
  },
  handler: async (ctx, { query: searchQuery, limit = 10, category }) => {
    let query = ctx.db
      .query("documents")
      .withSearchIndex("by_category", (q) =>
        q.search("category", searchQuery)
      );

    if (category) {
      query = query.filter((q) => q.eq(q.field("category"), category));
    }

    const results = await query.take(limit);

    return results.map(doc => ({
      ...doc,
      score: doc._score || 0.5, // Convex provides search score
    }));
  },
});

// Hybrid search combining vector + FTS
export const hybridSearch = query({
  args: {
    query: v.string(),
    limit: v.optional(v.number()),
    category: v.optional(v.string()),
  },
  handler: async (ctx, { query: searchQuery, limit = 10, category }) => {
    // 1. Generate embedding for query (requires Action)
    // Note: This needs to be called from an Action that generates the embedding first
    // For now, accept pre-computed embedding

    // Get embedding from query text via separate embedding generation
    const embedding = await generateQueryEmbedding(searchQuery);

    // 2. Run both searches in parallel
    const [vectorResults, ftsResults] = await Promise.all([
      ctx.runQuery(api.documents.vectorSearch, { embedding, limit: limit * 2, category }),
      ctx.runQuery(api.documents.fullTextSearch, { query: searchQuery, limit: limit * 2, category }),
    ]);

    // 3. Merge and deduplicate
    const merged = new Map<string, { doc: any; vectorScore: number; ftsScore: number }>();

    for (const r of vectorResults) {
      merged.set(r._id, { doc: r, vectorScore: r.score, ftsScore: 0 });
    }

    for (const r of ftsResults) {
      const existing = merged.get(r._id);
      if (existing) {
        existing.ftsScore = r.score;
      } else {
        merged.set(r._id, { doc: r, vectorScore: 0, ftsScore: r.score });
      }
    }

    // 4. Calculate combined score (weighted)
    const results = Array.from(merged.values()).map(({ doc, vectorScore, ftsScore }) => ({
      ...doc,
      score: (vectorScore * 0.7) + (ftsScore * 0.3), // Weight vector higher
    }));

    // 5. Sort by combined score and return top N
    return results
      .sort((a, b) => b.score - a.score)
      .slice(0, limit);
  },
});

// Helper: Cosine similarity
function cosineSimilarity(a: number[], b: number[]): number {
  if (a.length !== b.length) return 0;

  let dotProduct = 0;
  let normA = 0;
  let normB = 0;

  for (let i = 0; i < a.length; i++) {
    dotProduct += a[i] * b[i];
    normA += a[i] * a[i];
    normB += b[i] * b[i];
  }

  return dotProduct / (Math.sqrt(normA) * Math.sqrt(normB));
}
```
