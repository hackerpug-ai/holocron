## DESIGN
### Pattern (Iteration Summary Card)
```typescript
// components/research/IterationSummaryCard.tsx
import { View, StyleSheet } from 'react-native';
import { Text, Card } from 'react-native-paper';
import { useSemanticTheme } from '@/hooks/use-semantic-theme';
import { WorkerIndicator } from './WorkerIndicator';

interface IterationSummaryCardProps {
  iterationNumber: number;
  status: 'running' | 'completed' | 'error';
  coverageScore?: number;
  workers: Array<{
    id: string;
    name: string;
    status: 'pending' | 'running' | 'completed' | 'error';
    sourcesFound: number;
  }>;
  keyFindings?: string[];
  costSavings?: string; // "67% saved vs GPT-5"
}

export function IterationSummaryCard({
  iterationNumber,
  status,
  coverageScore,
  workers,
  keyFindings,
  costSavings,
}: IterationSummaryCardProps) {
  const { semantic } = useSemanticTheme();

  return (
    <Card style={[
      styles.container,
      {
        backgroundColor: semantic.color.card.default,
        borderColor: semantic.color.border.default,
        borderRadius: semantic.radius.lg,
      }
    ]}>
      <View style={styles.header}>
        <Text variant="titleMedium" style={{ color: semantic.color.onSurface.default }}>
          Iteration {iterationNumber}
        </Text>
        {coverageScore && (
          <CoverageBadge score={coverageScore} />
        )}
      </View>

      {/* Parallel Workers Section */}
      <View style={styles.workersSection}>
        <Text variant="labelMedium" style={{ color: semantic.color.onSurface.muted }}>
          Research Workers (GPT-5-mini)
        </Text>
        <View style={styles.workersList}>
          {workers.map((worker) => (
            <WorkerIndicator key={worker.id} worker={worker} />
          ))}
        </View>
      </View>

      {/* Key Findings */}
      {keyFindings && keyFindings.length > 0 && (
        <View style={styles.findingsSection}>
          <Text variant="labelMedium" style={{ color: semantic.color.onSurface.muted }}>
            Key Findings
          </Text>
          {keyFindings.slice(0, 3).map((finding, i) => (
            <Text key={i} variant="bodySmall" style={{ color: semantic.color.onSurface.default }}>
              • {finding}
            </Text>
          ))}
        </View>
      )}

      {/* Cost Savings Indicator */}
      {costSavings && (
        <View style={styles.costSection}>
          <Text variant="labelSmall" style={{ color: semantic.color.success.default }}>
            {costSavings} cost savings
          </Text>
        </View>
      )}
    </Card>
  );
}

// components/research/WorkerIndicator.tsx
export function WorkerIndicator({ worker }: { worker: WorkerStatus }) {
  const { semantic } = useSemanticTheme();

  const statusColor = {
    pending: semantic.color.onSurface.muted,
    running: semantic.color.info.default,
    completed: semantic.color.success.default,
    error: semantic.color.danger.default,
  }[worker.status];

  return (
    <View style={styles.workerContainer}>
      <View style={[styles.statusDot, { backgroundColor: statusColor }]} />
      <Text variant="bodySmall" style={{ color: semantic.color.onSurface.default }}>
        {worker.name}
      </Text>
      <Text variant="labelSmall" style={{ color: semantic.color.onSurface.muted }}>
        {worker.sourcesFound} sources
      </Text>
    </View>
  );
}
```

### Anti-pattern (DO NOT)
```typescript
// BAD: Hardcoded colors
backgroundColor: '#FFFFFF'  // Use semantic.color.card.default

// BAD: Not showing parallel nature
// Should show all workers, not just overall progress

// BAD: Not showing cost benefit
// Must highlight GPT-5-mini savings

// BAD: Raw subagent output
<Text>{subagentOutput}</Text>  // Summarize instead
```
