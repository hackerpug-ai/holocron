### Code Pattern
```typescript
import { View, StyleSheet } from 'react-native'
import { Text, Badge } from 'react-native-paper'
import { useSemanticTheme } from '@/hooks/use-semantic-theme'

interface ArticleDetailProps {
  article: MockArticle
  onClose: () => void
  visible: boolean
}

export function ArticleDetail({ article, onClose, visible }: ArticleDetailProps) {
  const { semantic } = useSemanticTheme()

  if (!visible) return null

  return (
    <View style={[styles.overlay, { backgroundColor: semantic.color.background.default }]}>
      <View style={styles.header}>
        <View style={styles.headerRow}>
          <Text variant="titleLarge" style={{ color: semantic.color.onSurface.default }}>
            {article.title}
          </Text>
        </View>
        <View style={styles.metaRow}>
          <Badge>{article.category}</Badge>
          <Text variant="labelSmall" style={{ color: semantic.color.onSurface.muted }}>
            {article.date}
          </Text>
        </View>
        <Pressable testID="article-detail-close" onPress={onClose}>
          <Text>Close</Text>
        </Pressable>
      </View>
      <ScrollView style={styles.content}>
        <Text variant="bodyMedium" style={{ color: semantic.color.onSurface.default }}>
          {article.content}
        </Text>
      </ScrollView>
    </View>
  )
}
```
