### Card Variants
1. **article** - Title, category badge, snippet, relevance score
2. **stats** - Total count, category breakdown, recent documents
3. **category_list** - Category names with document counts
4. **no_results** - Message when search returns empty
5. **category_not_found** - Error message with valid category list

### Interaction Notes
- Cards are tappable (onPress prop)
- Category badge uses semantic color mapping
- Snippet truncates at 150 chars with ellipsis
- Relevance score displayed as percentage (0-100)

### Code Pattern

```typescript
// components/ui/result-card.tsx
import { Pressable, StyleSheet, View } from 'react-native'
import { Text, Badge } from 'react-native-paper'
import { useSemanticTheme } from '@/hooks/use-semantic-theme'

interface ResultCardProps {
  cardType: 'article' | 'stats' | 'category_list' | 'no_results' | 'category_not_found'
  data: any
  onPress?: () => void
  testID?: string
}

export function ResultCard({ cardType, data, onPress, testID }: ResultCardProps) {
  const { semantic } = useSemanticTheme()

  if (cardType === 'article') {
    return (
      <Pressable
        testID={testID || 'result-card-article'}
        onPress={onPress}
        style={[styles.card, { backgroundColor: semantic.color.card.default }]}
      >
        <View style={styles.header}>
          <Text variant="titleMedium" style={{ color: semantic.color.onSurface.default }}>
            {data.title}
          </Text>
          <Badge>{data.category}</Badge>
        </View>
        <Text variant="bodyMedium" style={{ color: semantic.color.onSurface.muted }}>
          {data.snippet}
        </Text>
        {data.metadata?.relevance_score && (
          <Text variant="labelSmall" style={{ color: semantic.color.primary.default }}>
            {Math.round(data.metadata.relevance_score * 100)}% match
          </Text>
        )}
      </Pressable>
    )
  }

  // ... other card types
}

const styles = StyleSheet.create({
  card: {
    padding: 16,
    borderRadius: 12,
    borderWidth: 1,
    marginVertical: 4,
  },
  header: {
    flexDirection: 'row',
    justifyContent: 'space-between',
    alignItems: 'center',
    marginBottom: 8,
  },
})
```
