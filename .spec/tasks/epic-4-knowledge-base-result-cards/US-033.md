### Code Pattern
```typescript
import { View, ScrollView, StyleSheet } from 'react-native'
import { Text, TextInput, Badge } from 'react-native-paper'
import { useSemanticTheme } from '@/hooks/use-semantic-theme'

interface ArticlesScreenProps {
  articles: MockArticle[]
  selectedCategory?: string
  onCategoryChange: (category?: string) => void
  onSearch: (query: string) => void
  onArticlePress: (id: number) => void
}

export function ArticlesScreen({
  articles,
  selectedCategory,
  onCategoryChange,
  onSearch,
  onArticlePress
}: ArticlesScreenProps) {
  const { semantic } = useSemanticTheme()

  const categories = ['All', 'architecture', 'business', ...]

  return (
    <View style={[styles.container, { backgroundColor: semantic.color.background.default }]}>
      <ScrollView horizontal style={styles.chips} showsHorizontalScrollIndicator={false}>
        {categories.map(cat => (
          <Badge
            key={cat}
            testID={`articles-chip-${cat}`}
            onPress={() => onCategoryChange(cat === 'All' ? undefined : cat)}
            style={selectedCategory === cat || (!selectedCategory && cat === 'All')
              ? { backgroundColor: semantic.color.primary.default }
              : { backgroundColor: semantic.color.muted.default }
            }
          >
            {cat}
          </Badge>
        ))}
      </ScrollView>

      <TextInput
        testID="articles-search-input"
        placeholder="Search articles..."
        onChangeText={onSearch}
        style={styles.search}
      />

      <ScrollView>
        {articles.map(article => (
          <Pressable
            key={article.id}
            testID={`articles-card-${article.id}`}
            onPress={() => onArticlePress(article.id)}
          >
            <Text variant="titleMedium">{article.title}</Text>
            <Badge>{article.category}</Badge>
            <Text variant="bodySmall">{article.date}</Text>
          </Pressable>
        ))}
      </ScrollView>
    </View>
  )
}
```
