```typescript
// hooks/use-articles.ts
import { useQuery } from '@supabase-cache-helpers/react-query'
import { supabase } from '@/lib/supabase'

export function useArticles(category?: string, search?: string) {
  return useQuery(
    ['articles', category, search],
    async () => {
      let query = supabase
        .from('documents')
        .select('id, title, category, date, content')
        .order('created_at', { ascending: false })

      if (category) {
        query = query.eq('category', category)
      }

      if (search) {
        query = query.or(`title.ilike.%${search}%,content.ilike.%${search}%`)
      }

      const { data, error } = await query
      if (error) throw error
      return data
    }
  )
}

// In ArticlesScreen
const { data: articles, isLoading, error } = useArticles(selectedCategory, searchQuery)
```
