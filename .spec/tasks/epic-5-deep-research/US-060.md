### Code Pattern

See article-detail screen pattern

```typescript
// app/(app)/research/[sessionId].tsx
export default function ResearchDetailScreen() {
  const { sessionId } = useLocalSearchParams()
  const { session, loading, error } = useDeepResearchSession(sessionId)

  if (loading) return <LoadingSpinner />
  if (error) return <ErrorState message={error.message} />

  return <DeepResearchDetailView session={session} onBack={router.back} />
}
```
