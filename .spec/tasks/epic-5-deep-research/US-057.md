### Code Pattern

Source: `brain/skills/ralph-loop/SKILL.md`

```typescript
// Ralph Loop: iterate → review → refine → repeat until coverage >= 4 or max iterations
export async function runDeepResearchSession(sessionId: string) {
  let coverage = 0
  let iteration = 0

  while (coverage < 4 && iteration < maxIterations && !isCancelled(sessionId)) {
    iteration++

    const findings = await searchIteration(topic, refinedQueries)
    const feedback = await reviewFindings(findings)
    coverage = feedback.coverageScore

    await saveIteration({ sessionId, iteration, coverage, feedback })
    await streamIterationCard({ sessionId, iteration, coverage })

    if (coverage < 4) {
      refinedQueries = feedback.refinedQueries
    }
  }

  await completeSession(sessionId)
}
```

### Anti-pattern (DO NOT)

Do NOT process all iterations before showing any progress
