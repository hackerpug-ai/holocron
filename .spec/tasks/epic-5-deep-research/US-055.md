```typescript
// See generateStatsResponse pattern in slash-commands.ts
export async function handleDeepResearchCommand(input: string) {
  const { topic, maxIterations, type } = parseDeepResearchCommand(input)
  const sessionId = crypto.randomUUID()

  await createSession({ sessionId, topic, maxIterations, type })

  return {
    card_type: 'deep_research_confirmation',
    card_data: { topic, maxIterations: maxIterations || 5, sessionId }
  }
}
```
