```typescript
// See generateStatsResponse pattern in slash-commands.ts
export async function handleResumeCommand(input: string, userId: string) {
  const { sessionId, restart } = parseResumeCommand(input)

  if (sessionId) {
    return await resumeSession(sessionId, restart)
  }

  const sessions = await getIncompleteSessions(userId)
  return {
    card_type: 'resume_session_list',
    card_data: { sessions }
  }
}
```
