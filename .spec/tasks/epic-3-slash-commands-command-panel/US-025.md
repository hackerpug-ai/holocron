```typescript
// In supabase/functions/chat-send/slash-commands.ts
export function generateHelpResponse(): string {
  const header = 'Available commands:\n\n'
  const commands = SUPPORTED_COMMANDS.map((cmd) => {
    const syntax = cmd.syntax ? ` ${cmd.syntax}` : ''
    return `/${cmd.name}${syntax} - ${cmd.description}`
  }).join('\n')
  return header + commands
}

// In supabase/functions/chat-send/index.ts
import { parseSlashCommand, isKnownCommand, generateHelpResponse } from './slash-commands.ts'

// After inserting user message, check for /help
const parsed = parseSlashCommand(body.content)

let agentContent: string

if (parsed.isCommand && parsed.command === 'help') {
  // /help command - return command list
  agentContent = generateHelpResponse()
} else if (parsed.isCommand && isKnownCommand(parsed.command)) {
  // Known command - placeholder response (full implementation in later epics)
  agentContent = `Command /${parsed.command} recognized. Full implementation coming soon!`
} else if (parsed.isCommand) {
  // Unknown command
  agentContent = `Unknown command: /${parsed.command}. Type /help to see available commands.`
} else {
  // Regular message - use existing response logic
  agentContent = generateAgentResponse(body.content)
}

// Insert agent message
const { data: agentMessage, error: agentError } = await supabase
  .from('chat_messages')
  .insert({
    conversation_id: body.conversation_id,
    role: 'agent' as MessageRole,
    content: agentContent,
    message_type: 'text' as MessageType,
  })
  .select()
  .single()
```
