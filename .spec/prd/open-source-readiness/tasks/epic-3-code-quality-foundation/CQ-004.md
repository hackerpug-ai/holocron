### Code Pattern
```typescript
import { defineTestApp, TestApp } from "convex-test";
import { api } from "../../convex/_generated/api";

describe("Conversation CRUD", () => {
  let testApp: TestApp;

  beforeEach(() => {
    testApp = defineTestApp(api);
  });

  describe("Create", () => {
    it("creates a new conversation with title", async () => {
      const id = await testApp.run(api.conversations.create, {
        title: "My Conversation",
      });

      const conversation = await testApp.run(api.conversations.get, { id });
      expect(conversation).toMatchObject({
        title: "My Conversation",
        status: "active",
      });
    });

    it("requires a title to create conversation", async () => {
      await expect(
        testApp.run(api.conversations.create, {
          title: "",
        })
      ).rejects.toThrow();
    });
  });

  describe("Read", () => {
    it("lists all conversations for user", async () => {
      const id1 = await testApp.run(api.conversations.create, { title: "Chat 1" });
      const id2 = await testApp.run(api.conversations.create, { title: "Chat 2" });

      const conversations = await testApp.query(api.conversations.list);
      expect(conversations).toHaveLength(2);
      expect(conversations.map(c => c._id)).toContain(id1);
      expect(conversations.map(c => c._id)).toContain(id2);
    });

    it("returns null for non-existent conversation", async () => {
      const conversation = await testApp.run(api.conversations.get, {
        id: "non-existent" as any,
      });
      expect(conversation).toBeNull();
    });
  });

  describe("Update", () => {
    it("updates conversation title", async () => {
      const id = await testApp.run(api.conversations.create, {
        title: "Original Title",
      });

      await testApp.run(api.conversations.update, {
        id,
        title: "Updated Title",
      });

      const conversation = await testApp.run(api.conversations.get, { id });
      expect(conversation?.title).toBe("Updated Title");
    });
  });

  describe("Delete", () => {
    it("deletes conversation and its messages", async () => {
      const id = await testApp.run(api.conversations.create, { title: "To Delete" });
      await testApp.run(api.chat.send, {
        conversationId: id,
        content: "Test",
        role: "user",
      });

      await testApp.run(api.conversations.remove, { id });

      const conversation = await testApp.run(api.conversations.get, { id });
      expect(conversation).toBeNull();

      const messages = await testApp.query(api.chat.messages.list, {
        conversationId: id,
      });
      expect(messages).toHaveLength(0);
    });
  });
});
```
