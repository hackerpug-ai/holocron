```typescript
import { calculateConfidence } from "../../convex/research/confidence";

describe("Confidence Scoring", () => {
  describe("Edge Cases", () => {
    it("returns LOW when zero sources", () => {
      const result = calculateConfidence({
        sources: [],
        totalFindings: 0,
      });
      expect(result.level).toBe("LOW");
      expect(result.score).toBe(0);
    });

    it("returns LOW when all sources have low credibility", () => {
      const result = calculateConfidence({
        sources: [
          { credibility: 0.1, agreement: 0.5 },
          { credibility: 0.2, agreement: 0.3 },
        ],
        totalFindings: 5,
      });
      expect(result.level).toBe("LOW");
    });

    it("returns HIGH when all sources agree and have high credibility", () => {
      const result = calculateConfidence({
        sources: [
          { credibility: 0.9, agreement: 0.95 },
          { credibility: 0.95, agreement: 0.98 },
        ],
        totalFindings: 10,
      });
      expect(result.level).toBe("HIGH");
    });
  });

  describe("Boundary Conditions", () => {
    it("correctly handles HIGH/MEDIUM boundary", () => {
      // Test right at the threshold
      const highResult = calculateConfidence({
        sources: [{ credibility: 0.7, agreement: 0.7 }],
        totalFindings: 5,
      });
      expect(highResult.level).toBe("HIGH");

      const mediumResult = calculateConfidence({
        sources: [{ credibility: 0.6, agreement: 0.6 }],
        totalFindings: 5,
      });
      expect(mediumResult.level).toBe("MEDIUM");
    });

    it("correctly handles MEDIUM/LOW boundary", () => {
      const mediumResult = calculateConfidence({
        sources: [{ credibility: 0.4, agreement: 0.5 }],
        totalFindings: 2,
      });
      expect(mediumResult.level).toBe("MEDIUM");

      const lowResult = calculateConfidence({
        sources: [{ credibility: 0.3, agreement: 0.3 }],
        totalFindings: 1,
      });
      expect(lowResult.level).toBe("LOW");
    });
  });

  describe("Max Scores", () => {
    it("caps score at maximum threshold", () => {
      const result = calculateConfidence({
        sources: Array(20).fill({ credibility: 1.0, agreement: 1.0 }),
        totalFindings: 100,
      });
      expect(result.score).toBeLessThanOrEqual(1.0);
      expect(result.level).toBe("HIGH");
    });
  });

  describe("Agreement Weighting", () => {
    it("penalizes low agreement even with high credibility", () => {
      const disagreeingResult = calculateConfidence({
        sources: [
          { credibility: 0.9, agreement: 0.3 },
          { credibility: 0.9, agreement: 0.3 },
        ],
        totalFindings: 5,
      });

      const agreeingResult = calculateConfidence({
        sources: [
          { credibility: 0.9, agreement: 0.9 },
          { credibility: 0.9, agreement: 0.9 },
        ],
        totalFindings: 5,
      });

      expect(disagreeingResult.level).not.toBe("HIGH");
      expect(agreeingResult.level).toBe("HIGH");
    });
  });
});
```
