```typescript
// BEFORE — Bad (string-grep test)
import fs from 'fs';
import { VoiceSessionOverlay } from './VoiceSessionOverlay';

describe('VoiceSessionOverlay', () => {
  it('uses VoiceProvider', () => {
    const source = fs.readFileSync('./VoiceSessionOverlay.tsx', 'utf-8');
    expect(source).toContain('VoiceProvider');
  });
});

// AFTER — Good (component render test)
import { render, screen } from '@testing-library/react-native';
import { VoiceSessionOverlay } from './VoiceSessionOverlay';

describe('VoiceSessionOverlay', () => {
  it('renders voice session controls', () => {
    render(<VoiceSessionOverlay session={{ id: 'test' }} />);
    expect(screen.getByTestId('voice-session-overlay')).toBeTruthy();
  });

  it('displays recording state when active', () => {
    render(<VoiceSessionOverlay session={{ id: 'test', isRecording: true }} />);
    expect(screen.getByTestId('recording-indicator')).toBeTruthy();
  });

  it('calls onStop when stop button pressed', () => {
    const onStop = jest.fn();
    render(<VoiceSessionOverlay session={{ id: 'test' }} onStop={onStop} />);
    fireEvent.press(screen.getByTestId('stop-button'));
    expect(onStop).toHaveBeenCalled();
  });
});
```

## CODING STANDARDS

- **testing**: Render components, don't grep source files
- **quality**: Tests should verify behavior, not implementation details

## DEPENDENCIES

No dependencies.

## NOTES

- String-grep tests are extremely fragile — any refactoring breaks them
- They don't verify the component actually works, only that text exists
- The PRD specifically mentions `VoiceSessionOverlay.test.tsx` — check that file
- Search for patterns like `readFileSync`, `fs.readFile`, `toString()`
- If a component test file only has string-grep tests, delete the entire file
- If it has some good tests, just remove the bad ones
- Consider whether deleted tests need replacement with behavioral tests
