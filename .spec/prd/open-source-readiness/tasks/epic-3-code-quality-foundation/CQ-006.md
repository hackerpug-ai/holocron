### Code Pattern
```typescript
import { shouldTerminate, TerminationReason } from "../../convex/research/termination";

describe("Research Termination Logic", () => {
  describe("Max Iterations", () => {
    it("terminates when max iterations reached", () => {
      const result = shouldTerminate({
        iterationCount: 10,
        maxIterations: 10,
        elapsedTimeMs: 1000,
        timeoutMs: 5000,
        confidenceLevel: "MEDIUM",
      });
      expect(result.shouldTerminate).toBe(true);
      expect(result.reason).toBe(TerminationReason.MaxIterations);
    });

    it("does not terminate below max iterations", () => {
      const result = shouldTerminate({
        iterationCount: 5,
        maxIterations: 10,
        elapsedTimeMs: 1000,
        timeoutMs: 5000,
        confidenceLevel: "MEDIUM",
      });
      expect(result.shouldTerminate).toBe(false);
    });

    it("handles boundary at max iterations - 1", () => {
      const result = shouldTerminate({
        iterationCount: 9,
        maxIterations: 10,
        elapsedTimeMs: 1000,
        timeoutMs: 5000,
        confidenceLevel: "MEDIUM",
      });
      expect(result.shouldTerminate).toBe(false);
    });
  });

  describe("Timeout", () => {
    it("terminates when timeout exceeded", () => {
      const result = shouldTerminate({
        iterationCount: 3,
        maxIterations: 10,
        elapsedTimeMs: 6000,
        timeoutMs: 5000,
        confidenceLevel: "MEDIUM",
      });
      expect(result.shouldTerminate).toBe(true);
      expect(result.reason).toBe(TerminationReason.Timeout);
    });

    it("does not terminate before timeout", () => {
      const result = shouldTerminate({
        iterationCount: 3,
        maxIterations: 10,
        elapsedTimeMs: 4000,
        timeoutMs: 5000,
        confidenceLevel: "MEDIUM",
      });
      expect(result.shouldTerminate).toBe(false);
    });

    it("handles boundary exactly at timeout", () => {
      const result = shouldTerminate({
        iterationCount: 3,
        maxIterations: 10,
        elapsedTimeMs: 5000,
        timeoutMs: 5000,
        confidenceLevel: "MEDIUM",
      });
      expect(result.shouldTerminate).toBe(true);
    });
  });

  describe("Confidence Threshold", () => {
    it("terminates when HIGH confidence reached", () => {
      const result = shouldTerminate({
        iterationCount: 3,
        maxIterations: 10,
        elapsedTimeMs: 1000,
        timeoutMs: 5000,
        confidenceLevel: "HIGH",
      });
      expect(result.shouldTerminate).toBe(true);
      expect(result.reason).toBe(TerminationReason.HighConfidence);
    });

    it("does not terminate on MEDIUM confidence", () => {
      const result = shouldTerminate({
        iterationCount: 3,
        maxIterations: 10,
        elapsedTimeMs: 1000,
        timeoutMs: 5000,
        confidenceLevel: "MEDIUM",
      });
      expect(result.shouldTerminate).toBe(false);
    });

    it("does not terminate on LOW confidence", () => {
      const result = shouldTerminate({
        iterationCount: 3,
        maxIterations: 10,
        elapsedTimeMs: 1000,
        timeoutMs: 5000,
        confidenceLevel: "LOW",
      });
      expect(result.shouldTerminate).toBe(false);
    });
  });

  describe("Combined Criteria", () => {
    it("prioritizes timeout over max iterations", () => {
      const result = shouldTerminate({
        iterationCount: 10,
        maxIterations: 10,
        elapsedTimeMs: 6000,
        timeoutMs: 5000,
        confidenceLevel: "HIGH",
      });
      // Should terminate due to timeout first
      expect(result.reason).toBe(TerminationReason.Timeout);
    });

    it("prioritizes high confidence over other criteria", () => {
      const result = shouldTerminate({
        iterationCount: 3,
        maxIterations: 10,
        elapsedTimeMs: 1000,
        timeoutMs: 5000,
        confidenceLevel: "HIGH",
      });
      expect(result.reason).toBe(TerminationReason.HighConfidence);
    });
  });
});
```
