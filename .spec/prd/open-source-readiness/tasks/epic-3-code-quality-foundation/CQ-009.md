## DESIGN

### References
- PRD Section 2.6, AC-36

### Code Pattern
```typescript
// BEFORE — Noisy console.log
export const myFunction = async (ctx: MutationCtx, args: { id: string }) => {
  console.log("myFunction called with", args);
  const result = await ctx.db.get(args.id);
  console.log("Got result:", result);
  // ...
  console.log("Returning from myFunction");
};

// AFTER — Structured logging (only essential)
import { logEvent } from "../lib/logging";

export const myFunction = async (ctx: MutationCtx, args: { id: string }) => {
  logEvent("myFunction_invoked", { id: args.id });
  const result = await ctx.db.get(args.id);
  // ...
  return result;
};

// OR — Use Convex's built-in logging (if available)
import { v } from "convex/values";

export const myFunction = mutation({
  args: { id: v.id() },
  handler: async (ctx, args) => {
    // Use ctx.logger if available
    ctx.logger?.info("myFunction called", { id: args.id });
    // ...
  },
});
```
