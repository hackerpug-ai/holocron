### Code Pattern
```typescript
import { defineTestApp, TestApp } from "convex-test";
import { api } from "../../convex/_generated/api";

describe("Chat Send Flow", () => {
  let testApp: TestApp;

  beforeEach(() => {
    testApp = defineTestApp(api);
  });

  it("persists user message to database", async () => {
    const conversationId = await testApp.run(api.conversations.create, {
      title: "Test Chat",
    });

    await testApp.run(api.chat.send, {
      conversationId,
      content: "Hello, AI!",
      role: "user",
    });

    const messages = await testApp.query(api.chat.messages.list, {
      conversationId,
    });
    expect(messages).toHaveLength(1);
    expect(messages[0].content).toBe("Hello, AI!");
    expect(messages[0].role).toBe("user");
  });

  it("generates AI response after user message", async () => {
    const conversationId = await testApp.run(api.conversations.create, {
      title: "Test Chat",
    });

    await testApp.run(api.chat.send, {
      conversationId,
      content: "Hello, AI!",
      role: "user",
    });

    // Wait for AI response (may need to poll or use a hook)
    const messages = await testApp.query(api.chat.messages.list, {
      conversationId,
    });
    expect(messages.length).toBeGreaterThanOrEqual(2);
    expect(messages[1].role).toBe("assistant");
  });

  it("stores message with correct metadata", async () => {
    const conversationId = await testApp.run(api.conversations.create, {
      title: "Test Chat",
    });

    await testApp.run(api.chat.send, {
      conversationId,
      content: "Test message",
      role: "user",
    });

    const messages = await testApp.query(api.chat.messages.list, {
      conversationId,
    });
    expect(messages[0]).toMatchObject({
      content: "Test message",
      role: "user",
      conversationId,
      expect.objectContaining({ createdAt: expect.any(Number) }),
    });
  });
});
```
