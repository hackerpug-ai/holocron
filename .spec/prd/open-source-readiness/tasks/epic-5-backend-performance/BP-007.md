```typescript
// BEFORE — Loads all iterations, filters in JS (SLOW!)
export const fullTextSearchIterations = query({
  args: {
    sessionId: v.optional(v.id("researchSessions")),
    searchQuery: v.string(),
  },
  handler: async (ctx, args) => {
    // ❌ Loads ALL iterations across ALL sessions
    let iterations = await ctx.db.query("researchIterations").collect();

    // Then filters in JavaScript
    if (args.sessionId) {
      iterations = iterations.filter(i => i.sessionId === args.sessionId);
    }

    // Then filters by search text in JavaScript
    iterations = iterations.filter(i =>
      i.content.toLowerCase().includes(args.searchQuery.toLowerCase())
    );

    return iterations;
  },
});

// AFTER — Uses index for sessionId (FAST!)
export const fullTextSearchIterations = query({
  args: {
    sessionId: v.optional(v.id("researchSessions")),
    searchQuery: v.string(),
  },
  handler: async (ctx, args) => {
    let query = ctx.db.query("researchIterations");

    // ✅ Use the index for sessionId filtering
    if (args.sessionId) {
      query = query.withIndex("by_session", q =>
        q.eq("sessionId", args.sessionId)
      );
    }

    // For text search, use withSearchIndex if available
    // Otherwise, collect smaller result set and filter in JS
    const iterations = await query.collect();

    // Filter by search text (still in JS, but on much smaller set)
    return iterations.filter(i =>
      i.content.toLowerCase().includes(args.searchQuery.toLowerCase())
    );
  },
});

// EVEN BETTER — Use search index for text too
export const fullTextSearchIterations = query({
  args: {
    sessionId: v.optional(v.id("researchSessions")),
    searchQuery: v.string(),
  },
  handler: async (ctx, args) => {
    // Use search index if available
    if (args.sessionId) {
      // Combine session index + search
      return await ctx.db.query("researchIterations")
        .withIndex("by_session", q => q.eq("sessionId", args.sessionId))
        .filter(q => q.eq("content", args.searchQuery)) // If search index exists
        .collect();
    } else {
      // Full text search across all iterations
      return await ctx.db.query("researchIterations")
        .withSearchIndex("by_content", q => q.search("content", args.searchQuery))
        .collect();
    }
  },
});
```
