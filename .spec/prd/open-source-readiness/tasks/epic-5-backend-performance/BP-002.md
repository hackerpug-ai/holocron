```typescript
// BEFORE — Manual JS cosine similarity (VERY SLOW)
export const findSimilarDocuments = query({
  args: { embedding: v.array(v.number()) },
  handler: async (ctx, args) => {
    const allDocs = await ctx.db.query("documents").collect(); // Loads everything

    // Manual cosine similarity in JS
    const similarities = allDocs.map(doc => {
      const similarity = cosineSimilarity(args.embedding, doc.embedding);
      return { doc, similarity };
    });

    // Sort by similarity
    return similarities
      .sort((a, b) => b.similarity - a.similarity)
      .slice(0, 10)
      .map(s => s.doc);
  },
});

// AFTER — Native vector search (FAST)
export const findSimilarDocuments = query({
  args: { embedding: v.array(v.number()) },
  handler: async (ctx, args) => {
    // Native Convex vector search — uses the index, optimized
    const results = await ctx.vectorSearch("documents", "by_embedding", {
      vector: args.embedding,
      limit: 10,
    });

    return results.map(r => ({
      _id: r._id,
      title: r.title,
      content: r.content,
      score: r.score, // Similarity score included
    }));
  },
});
```
