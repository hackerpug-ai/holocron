### Code Pattern
```typescript
// BEFORE — Unnecessary read-after-write
export const updateResearchSession = mutation({
  args: { id: v.id("researchSessions"), status: v.string() },
  handler: async (ctx, args) => {
    await ctx.db.patch(args.id, { status: args.status });

    // ❌ Unnecessary — patch already succeeded
    const updated = await ctx.db.get(args.id);
    if (!updated) {
      throw new Error("Session not found");
    }

    return updated;
  },
});

// AFTER — Just patch, no redundant read
export const updateResearchSession = mutation({
  args: { id: v.id("researchSessions"), status: v.string() },
  handler: async (ctx, args) => {
    // ✅ Atomic operation — if it succeeds, the write happened
    await ctx.db.patch(args.id, { status: args.status });

    // Return success, no need to re-read
    return { success: true };
  },
});

// EXCEPTION — If you NEED the updated document:
export const updateAndReturnSession = mutation({
  args: { id: v.id("researchSessions"), status: v.string() },
  handler: async (ctx, args) => {
    // ✅ If you need to return the document, do the read BEFORE
    const session = await ctx.db.get(args.id);
    if (!session) {
      throw new Error("Session not found");
    }

    await ctx.db.patch(args.id, { status: args.status });

    // Return the session you already read (with the patch applied implicitly)
    return { ...session, status: args.status };
  },
});
```
