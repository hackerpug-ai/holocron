// BEFORE — Full table scan (slow!)
export const searchDocuments = query({
  args: { query: v.string() },
  handler: async (ctx, args) => {
    const allDocs = await ctx.db.query("documents").collect(); // ❌ Scans entire table
    return allDocs.filter(doc =>
      doc.title.toLowerCase().includes(args.query.toLowerCase()) ||
      doc.content.toLowerCase().includes(args.query.toLowerCase())
    );
  },
});

// AFTER — Indexed search (fast!)
export const searchDocuments = query({
  args: { query: v.string() },
  handler: async (ctx, args) => {
    // Use the search index defined in schema
    const results = await ctx.db.query("documents")
      .withSearchIndex("by_category", (searcher) =>
        searcher.search("title", args.query)
      )
      .take(20);
    return results;
  },
});

// For vector search:
// BEFORE — Manual cosine similarity over all docs
const allDocs = await ctx.db.query("documents").collect();
const similarities = allDocs.map(doc => ({

// AFTER — Native vector search
const results = await ctx.vectorSearch("documents", "by_embedding", {
  vector: queryEmbedding,

});
