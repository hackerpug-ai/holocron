### If It's a Wrapper
```typescript
// ❌ If this is all it does, delete it:
export function useRouting() {
  return useRouter()
}

// ✅ Use expo-router directly:
import { useRouter } from 'expo-router'

const router = useRouter()
router.push('/path')
```

### If It Adds Value
```typescript
// ✅ Keep if it adds real functionality:
export function useRouting() {
  const router = useRouter()

  return {
    ...router,
    // Custom methods that add value
    goBack: () => {
      if (canGoBack()) {
        router.back()
      } else {
        router.replace('/home')
      }
    },
  }
}
```
