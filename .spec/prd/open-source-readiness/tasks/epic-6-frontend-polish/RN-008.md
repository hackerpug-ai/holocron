### Anti-Pattern
```typescript
// ❌ WRONG: State syncing with useEffect
export function useScroll() {
  const [scrollY, setScrollY] = useState(0)

  useEffect(() => {
    // This doesn't work — you can't sync scroll position with effects
    const scrollView = document.querySelector('scrollview')
    setScrollY(scrollView?.scrollY ?? 0)
  }, [])

  return { scrollY }
}
```

### Correct Pattern
```typescript
// ✅ GOOD: Direct scroll event handling
import { useRef, useState } from 'react'
import { ScrollView, ScrollViewProps } from 'react-native'

export function useScroll() {
  const scrollViewRef = useRef<ScrollView>(null)
  const [scrollY, setScrollY] = useState(0)

  const handleScroll: ScrollViewProps['onScroll'] = (event) => {
    setScrollY(event.nativeEvent.contentOffset.y)
  }

  return {
    scrollViewRef,
    scrollY,
    onScroll: handleScroll,
    scrollEventThrottle: 16, // ~60fps
  }
}

// ✅ BETTER: Using Animated API for performance
import Animated from 'react-native-reanimated'

export function useAnimatedScroll() {
  const scrollY = useRef(new Animated.Value(0)).current

  const onScroll = Animated.event(
    [{ nativeEvent: { contentOffset: { y: scrollY } } }],
    { useNativeDriver: true }
  )

  return { scrollY, onScroll }
}
```

## CODING STANDARDS

- **hooks**: Events are source of truth, not derived state
- **performance**: Use Animated API for smooth scrolling

## DEPENDENCIES

No dependencies.

## NOTES

- Scroll position should come from onScroll events, not effects
- For smooth animations, use React Native Reanimated
- The native driver option is important for performance
- scrollEventThrottle controls how often events fire (16ms = ~60fps)
- This is a common anti-pattern in React Native
