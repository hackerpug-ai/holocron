### Correct Pattern
```typescript
// ✅ GOOD: React Native testID
<Button testID="login-screen-submit-button" onPress={handleSubmit}>
  Submit
</Button>

<TextInput
  testID="login-screen-email-input"
  value={email}
  onChangeText={setEmail}
/>

<Pressable testID="card-pressable" onPress={handlePress}>
  <Text>Card</Text>
</Pressable>

// ✅ GOOD: Web data-testid
<button
  data-testid="login-screen-submit-button"
  onClick={handleSubmit}
>
  Submit
</button>

<input
  data-testid="login-screen-email-input"
  value={email}
  onChange={(e) => setEmail(e.target.value)}
/>

// ❌ BAD: Missing test IDs
<Button onPress={handleSubmit}>Submit</Button>
<input value={email} onChange={...} />
```

### Finding Issues
```bash
# Find buttons without test IDs
grep -rn "<Button" app/ components/ | grep -v "testID\|data-testid"

# Find inputs without test IDs
grep -rn "<TextInput\|<input" app/ components/ | grep -v "testID\|data-testid"

# Find pressables without test IDs
grep -rn "<Pressable" app/ | grep -v "testID"
```
