### Finding Missing Stories
```bash
# Find all UI components
find components/ui -name "*.tsx" -not -name "*.stories.tsx"

# Check which have stories
for file in components/ui/*.tsx; do
  if [ ! -f "${file%.tsx}.stories.tsx" ]; then
    echo "$file is missing stories"
  fi
done
```

### Story Template
```typescript
import type { Meta, StoryObj } from '@storybook/react'
import { ComponentName } from './component-name'
import { within, userEvent, expect } from '@storybook/test'

const meta = {
  title: 'UI/ComponentName',
  component: ComponentName,
} satisfies Meta<typeof ComponentName>

export default meta
type Story = StoryObj<typeof meta>

export const Default: Story = {
  args: {
    // Minimal props
  },
}

// REQUIRED if component has variants
export const AllVariants: Story = {
  render: () => (
    <div style={{ display: 'flex', gap: 16, flexWrap: 'wrap' }}>
      <ComponentName variant="primary" />
      <ComponentName variant="secondary" />
      <ComponentName variant="tertiary" />
    </div>
  ),
}

// REQUIRED for interactive components
export const Interactive: Story = {
  play: async ({ canvasElement }) => {
    const canvas = within(canvasElement)
    const button = canvas.getByRole('button')
    await userEvent.click(button)
    await expect(button).toBeChecked()
  },
}
```

### Story Structure
1. **Default** - Basic usage with minimal props
2. **AllVariants** - Grid showing all component variants
3. **Interactive** - Play function testing interactions (if applicable)
4. **States** - Loading, error, disabled states (if applicable)
