### Correct Pattern
```typescript
import { useState, useEffect } from 'react'

export function useDebounce<T>(value: T, delay: number): T {
  const [debouncedValue, setDebouncedValue] = useState<T>(value)

  useEffect(() => {
    // Set timeout to update debounced value
    const timeout = setTimeout(() => {
      setDebouncedValue(value)
    }, delay)

    // Cleanup: clear timeout if value changes or unmounts
    return () => {
      clearTimeout(timeout)
    }
  }, [value, delay])

  return debouncedValue
}

// Usage:
const debouncedSearch = useDebounce(searchTerm, 300)
```

### Test Pattern
```typescript
import { renderHook, waitFor } from '@testing-library/react'

describe('useDebounce', () => {
  it('delays value updates', async () => {
    const { result, rerender } = renderHook(
      ({ value, delay }) => useDebounce(value, delay),
      { initialProps: { value: 'initial', delay: 300 } }
    )

    expect(result.current).toBe('initial')

    rerender({ value: 'updated', delay: 300 })

    // Should still be initial
    expect(result.current).toBe('initial')

    // After delay, should be updated
    await waitFor(
      () => expect(result.current).toBe('updated'),
      { timeout: 500 }
    )
  })
})
```

- Common issue: forgetting cleanup function in useEffect
- Without cleanup, you'll get memory leaks and warnings
- The delay should be in milliseconds (typical: 300-500ms)
