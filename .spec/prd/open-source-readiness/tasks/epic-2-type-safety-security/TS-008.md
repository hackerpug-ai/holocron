### Code Pattern
```typescript
// Before — In-memory (broken in serverless)
const rateLimiter = new Map<string, number>();

export const checkRateLimit = (key: string, limit: number) => {
  const count = rateLimiter.get(key) || 0;
  if (count >= limit) return false;
  rateLimiter.set(key, count + 1);
  return true;
};

// After — Database-backed
export const checkRateLimit = async (
  ctx: MutationCtx,
  key: string,
  limit: number,
  windowMs: number
): Promise<boolean> => {
  const now = Date.now();
  const windowStart = now - windowMs;

  // Clean old entries
  const oldEntries = await ctx.db
    .query("rateLimits")
    .withIndex("by_key", q => q.eq("key", key))
    .filter(q => q.lt(q.field("timestamp"), windowStart))
    .collect();
  for (const entry of oldEntries) {
    await ctx.db.delete(entry._id);
  }

  // Count current window
  const current = await ctx.db
    .query("rateLimits")
    .withIndex("by_key", q => q.eq("key", key))
    .collect();

  if (current.length >= limit) return false;

  // Record this request
  await ctx.db.insert("rateLimits", { key, timestamp: now });
  return true;
};
```

## CODING STANDARDS

- **serverless**: Never rely on in-memory state for persistence
- **convex**: Use documents for all persistent state

## NOTES

- Serverless functions can restart at any time — in-memory state is lost
- The reference implementation in `rateLimits.ts` shows the correct pattern
- You may need to add a `rateLimits` table to the schema
- Add an index on `key` and `timestamp` for efficient queries
- Consider cleanup of old entries to avoid bloat
