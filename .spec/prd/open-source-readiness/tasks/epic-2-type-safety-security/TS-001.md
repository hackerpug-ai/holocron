### Code Pattern
```typescript
// Before
import { ActionCtx } from "convex/server";

export const chatAction = async (ctx: any, args: any) => {
  const result = await llmCall();
  return result as any;
};

// After
import { ActionCtx } from "convex/server";
import type { CoreMessage } from "@ai-sdk/react";

export const chatAction = async (
  ctx: ActionCtx,
  args: { messages: CoreMessage[] }
): Promise<{ content: string; role: string }> => {
  const result = await llmCall(args.messages);
  return result;
};
```
