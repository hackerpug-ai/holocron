export const discoverCreator = {
  name: "discoverCreator",
  description: "Discover a creator by looking up their platforms",
  inputSchema: {
    type: "object",
    properties: {
      name: { type: "string", description: "Creator name or handle" },
      platformHints: {
        type: "object",
        description: "Optional platform-specific handles",
        properties: {
          youtube: { type: "string" },
          twitter: { type: "string" },
          bluesky: { type: "string" },
          github: { type: "string" },
        },
      },
    },
    required: ["name"],
  },
  handler: async (ctx: RequestContext, args: any) => {
    try {
      const result = await ctx.runAction(api.creators.actions.discover, {
        name: args.name,
        platformHints: args.platformHints,
      });

      return {
        success: true,
        data: result,
      };
    } catch (error) {
      return {
        success: false,
        error: error instanceof Error ? error.message : "Unknown error",
      };
    }
  },
};
```

// ADD to holocron-mcp/src/tools/creators.ts

/**
 * Assimilate a creator profile with transcript extraction
 * Fetches all videos from creator's channel and creates transcript jobs
 */
export const assimilateCreator = {
  name: "assimilateCreator",
  description: "Assimilate a creator by extracting transcripts from all their videos. Creates transcript jobs for up to maxVideos most recent videos.",
  inputSchema: {
    type: "object",
    properties: {
      profileId: {
        type: "string",
        description: "Creator profile ID (Convex ID)",
      },
      maxVideos: {
        type: "number",
        description: "Maximum number of videos to process (default: 50)",
      },
      includeTranscripts: {
        type: "boolean",
        description: "Whether to extract transcripts (default: true)",
      },
      forceRegenerate: {
        type: "boolean",
        description: "Re-transcribe videos that already have transcripts (default: false)",
      },
    },
    required: ["profileId"],
  },
  handler: async (ctx: RequestContext, args: any) => {
    try {
      const result = await ctx.runAction(api.creators.actions.assimilateCreator, {
        profileId: args.profileId as any, // Convex ID type
        options: {
          maxVideos: args.maxVideos ?? 50,
          includeTranscripts: args.includeTranscripts ?? true,
          forceRegenerate: args.forceRegenerate ?? false,
        },
      });

      return {
        success: true,
        data: {
          documentId: result.documentId,
          profileId: result.profileId,
          videosFound: result.videosFound,
          transcriptsCreated: result.transcriptsCreated,
          transcriptsSkipped: result.transcriptsSkipped,
          status: result.status,
        },
      };
    } catch (error) {
      return {
        success: false,
        error: error instanceof Error ? error.message : "Failed to assimilate creator",
      };
    }
  },
};

/**
 * Get all transcripts for a creator
 * Returns list of transcripts with preview text and metadata
 */
export const getCreatorTranscripts = {
  name: "getCreatorTranscripts",
  description: "Retrieve all transcripts for a creator profile. Returns transcript metadata including preview text, word count, and source.",
  inputSchema: {
    type: "object",
    properties: {
      profileId: {
        type: "string",
        description: "Creator profile ID (Convex ID)",
      },
      limit: {
        type: "number",
        description: "Maximum number of transcripts to return (default: 50)",
      },
    },
    required: ["profileId"],
  },
  handler: async (ctx: RequestContext, args: any) => {
    try {
      // First, get all videos for this creator
      const profile = await ctx.runQuery(api.creators.queries.get, {
        profileId: args.profileId as any,
      });

      if (!profile.creator) {
        return {
          success: false,
          error: "Creator profile not found",
        };
      }

      // Get transcripts for creator's videos
      const transcripts = await ctx.runQuery(api.transcripts.listByCreator, {
        profileId: args.profileId as any,
        limit: args.limit ?? 50,
      });

      return {
        success: true,
        data: {
          profileId: args.profileId,
          creatorHandle: profile.creator.handle,
          transcriptCount: transcripts.length,
          transcripts: transcripts.map(t => ({
            contentId: t.contentId,
            sourceUrl: t.sourceUrl,
            transcriptSource: t.transcriptSource,
            previewText: t.previewText,
            wordCount: t.wordCount,
            generatedAt: t.generatedAt,
          })),
        },
      };
    } catch (error) {
      return {
        success: false,
        error: error instanceof Error ? error.message : "Failed to get transcripts",
      };
    }
  },
};

/**
 * Regenerate a transcript for a specific video
 * Creates a new transcript job, re-processing the video
 */
export const regenerateTranscript = {
  name: "regenerateTranscript",
  description: "Force re-transcription of a video. Creates a new transcript job that will re-fetch the transcript using YouTube API or Jina Reader fallback.",
  inputSchema: {
    type: "object",
    properties: {
      contentId: {
        type: "string",
        description: "YouTube video ID",
      },
      sourceUrl: {
        type: "string",
        description: "YouTube video URL (optional, will be generated if not provided)",
      },
      priority: {
        type: "number",
        description: "Job priority (0-10, default: 5)",
      },
    },
    required: ["contentId"],
  },
  handler: async (ctx: RequestContext, args: any) => {
    try {
      const sourceUrl = args.sourceUrl || `https://www.youtube.com/watch?v=${args.contentId}`;

      const result = await ctx.runMutation(internal.transcripts.createTranscriptJob, {
        contentId: args.contentId,
        sourceUrl,
        priority: args.priority ?? 5,
      });

      return {
        success: true,
        data: {
          jobId: result.jobId,
          created: result.created,
          contentId: args.contentId,
          message: result.created
            ? "Transcript job created"
            : "Transcript job already exists",
        },
      };
    } catch (error) {
      return {
        success: false,
        error: error instanceof Error ? error.message : "Failed to create transcript job",
      };
    }
  },
};
```
