# US-X-002: Offline Feed Support

> Task ID: US-X-002
> Type: FEATURE
> Priority: P1
> Estimate: 120 minutes
> Assignee: frontend-designer

## CRITICAL CONSTRAINTS

### MUST
- Cache last 50 items locally
- View cached content offline
- Show "Offline" indicator when no connection
- Queue feedback for later submission
- Cache expires after 7 days

### NEVER
- Show stale content without indication
- Lose user feedback (queue it)
- Block UI when offline
- Cache sensitive data

### STRICTLY
- Use Convex local cache
- Use AsyncStorage for feedback queue
- Show clear offline indicator

## SPECIFICATION

**Objective:** Enable offline viewing of cached feed content with queued feedback submission

**Success looks like:** User opens app on airplane, sees last 50 items with "Offline" banner, can provide feedback that syncs when back online

## ACCEPTANCE CRITERIA

| # | Given | When | Then | Verify |

| 1 | User goes offline | Network lost | "Offline" banner appears at top | `getByTestId('offline-banner')` exists |

### Offline Detection Pattern

```typescript
// hooks/useNetworkStatus.ts
import { useEffect, useState } from 'react'
import { NetInfo } from 'react-native-netinfo'

export function useNetworkStatus() {
  const [isOnline, setIsOnline] = useState(true)
  
  useEffect(() => {
    const unsubscribe = NetInfo.addEventListener(state => {
      setIsOnline(state.isConnected ?? true)
    })
    return unsubscribe
  }, [])
  
  return { isOnline, isOffline: !isOnline }
}

// components/OfflineBanner.tsx
export function OfflineBanner() {
  const { isOffline } = useNetworkStatus()
  
  if (!isOffline) return null
  
  return (
    <View 
      className="bg-amber-500 p-2 items-center"
      testID="offline-banner"
    >
      <Text variant="labelMedium" className="text-amber-950">
        You're offline. Showing cached content.
      </Text>
    </View>
  )
}

// hooks/useOfflineQueue.ts
import AsyncStorage from '@react-native-async-storage/async-storage'

const FEEDBACK_QUEUE_KEY = 'offline_feedback_queue'

interface QueuedFeedback {
  findingId: string
  sentiment: 'positive' | 'negative'
  timestamp: number
}

export function useOfflineQueue() {
  const { isOnline } = useNetworkStatus()
  
  const queueFeedback = async (feedback: QueuedFeedback) => {
    const queue = await getQueue()
    queue.push(feedback)
    await AsyncStorage.setItem(FEEDBACK_QUEUE_KEY, JSON.stringify(queue))
  }
  
  const getQueue = async (): Promise<QueuedFeedback[]> => {
    const stored = await AsyncStorage.getItem(FEEDBACK_QUEUE_KEY)
    return stored ? JSON.parse(stored) : []
  }
  
  const flushQueue = async () => {
    if (!isOnline) return
    
    const queue = await getQueue()
    if (queue.length === 0) return
    
    // Submit all queued feedback
    await Promise.all(
      queue.map(f => submitFeedback(f.findingId, f.sentiment))
    )
    
    // Clear queue
    await AsyncStorage.removeItem(FEEDBACK_QUEUE_KEY)
  }
  
  // Flush queue when coming online
  useEffect(() => {
    if (isOnline) {
      flushQueue()
    }
  }, [isOnline])
  
  return { queueFeedback, getQueue }
}

// Usage in card
const handleFeedback = async (type: FeedbackType) => {
  if (isOnline) {
    await submitFeedback(finding.url, type)
  } else {
    await queueFeedback({ findingId: finding.url, sentiment: type, timestamp: Date.now() })
    // Show indicator
  }
}
```

### Cache Expiration

```typescript
// Check cache age
const CACHE_MAX_AGE = 7 * 24 * 60 * 60 * 1000 // 7 days

function isCacheValid(cachedAt: number): boolean {
  return Date.now() - cachedAt < CACHE_MAX_AGE
}

// When loading cached data
const cachedReport = await getCachedReport()
if (cachedReport && isCacheValid(cachedReport.cachedAt)) {
  return cachedReport
} else {
  // Clear stale cache
  await clearCache()
  // Fetch fresh data
  return await fetchReport()
}
```

### Anti-pattern (DO NOT)
```typescript
// ❌ WRONG: No offline indication
// User doesn't know they're seeing old content

// ❌ WRONG: Losing feedback
await submitFeedback() // Fails silently when offline

// ❌ WRONG: Never expiring cache
// User sees month-old content
```
