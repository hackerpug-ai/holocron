- Speak user-friendly error messages (no technical jargon)

## SPECIFICATION

**Objective:** Handle function call execution failures with appropriate classification, retry for transient errors, and spoken guidance for permanent errors

**Success looks like:** Transient errors silently retried; permanent errors explained to user in plain language; all errors logged for debugging

```typescript
type ErrorClass = 'transient' | 'permanent' | 'rate_limit';

function classifyError(error: unknown): ErrorClass {
  if (error instanceof Error) {
    if (error.message.includes('timeout') || error.message.includes('500')) return 'transient';
    if (error.message.includes('429')) return 'rate_limit';
  }
  return 'permanent';
}

function getSpokenErrorMessage(errorClass: ErrorClass, context?: string): string {
  switch (errorClass) {
    case 'transient': return 'Something went wrong. Let me try again.';
    case 'rate_limit': return 'Too many requests. Try again in a moment.';
    case 'permanent': return context
      ? `I couldn't ${context}.`
      : "I can't do that right now.";
  }
}
```
