### Code Pattern

```typescript
import { useAction, useMutation } from "convex/react";
import { api } from "../convex/_generated/api";

export function useVoiceSession(conversationId: Id<'conversations'>) {
  const [state, dispatch] = useReducer(voiceSessionReducer, initialState);
  const createSession = useAction(api.voice.actions.createSession);
  const endSession = useMutation(api.voice.mutations.endSession);
  const connectionRef = useRef<WebRTCConnection | null>(null);

  const start = async () => {
    dispatch({ type: 'CONNECT', conversationId });
    const { ephemeralKey, sessionId } = await createSession({ conversationId });
    const conn = new WebRTCConnection();
    await conn.connect(ephemeralKey);
    connectionRef.current = conn;
    dispatch({ type: 'CONNECTED', sessionId });
  };

  const stop = async () => {
    connectionRef.current?.destroy();
    await endSession({ sessionId: state.sessionId });
    dispatch({ type: 'DISCONNECT' });
  };

  useEffect(() => {
    return () => { stop(); };
  }, []);

  return { state, start, stop };
}
```

### Anti-pattern (DO NOT)
Do NOT use multiple useState. Do NOT skip useEffect cleanup.
