**Follow this exact screen component pattern:**

```typescript
// Location: screens/subscriptions-feed-screen.tsx (or components/subscriptions/SubscriptionFeedScreen.tsx)

import { useState } from 'react'
import { View, ScrollView, StyleSheet } from 'react-native'
import { useQuery } from 'convex/react'
import { api } from '@/convex/_generated/api'
import { SectionHeader } from '@/components/SectionHeader'
import { Settings } from '@/components/ui/icons'
import { SubscriptionFeedFilters } from '@/components/subscriptions/SubscriptionFeedFilters'
import { SubscriptionSettingsModal } from '@/components/subscriptions/SubscriptionSettingsModal'
import type { FilterType } from '@/components/subscriptions/SubscriptionFeedFilters'

export function SubscriptionFeedScreen() {
  const [selectedFilter, setSelectedFilter] = useState<FilterType>('all')
  const [settingsVisible, setSettingsVisible] = useState(false)

  // Fetch feed data (placeholder - actual query in EPIC-03)
  const { data: feedItems, isLoading } = useQuery(api.feeds.getFeed, {
    contentType: selectedFilter === 'all' ? undefined : selectedFilter,
  })

  const { data: counts } = useQuery(api.feeds.getCountsByType, {})

  return (
    <View style={styles.container} testID="subscription-feed-screen">
      {/* Header with settings button */}
      <SectionHeader
        title="Subscription Feed"
        actionLabel=""
        onActionPress={() => setSettingsVisible(true)}
        showChevron={false}
        size="lg"
      />

      {/* Filters */}
      <SubscriptionFeedFilters
        selectedFilter={selectedFilter}
        onFilterChange={setSelectedFilter}
        counts={counts ?? { all: 0, video: 0, blog: 0, social: 0 }}
        testID="feed-filters"
      />

      {/* Feed content */}
      <ScrollView style={styles.content}>
        {isLoading ? (
          <View style={styles.centered}>
            <Text>Loading feed...</Text>
          </View>
        ) : feedItems && feedItems.length > 0 ? (
          // Feed items will be rendered here (EPIC-05)
          feedItems.map((item) => (
            <View key={item._id} style={styles.placeholder}>
              <Text>{item.title}</Text>
            </View>
          ))
        ) : (
          <View style={styles.centered}>
            <Text>No items yet</Text>
          </View>
        )}
      </ScrollView>

      {/* Settings modal */}
      <SubscriptionSettingsModal
        visible={settingsVisible}
        onDismiss={() => setSettingsVisible(false)}
        onManageSubscriptions={() => {
          setSettingsVisible(false)
          // Navigate to management screen
          router.push('/subscriptions')
        }}
      />
    </View>
  )
}

const styles = StyleSheet.create({
  container: {
    flex: 1,
    backgroundColor: '#fff',
  },
  content: {
    flex: 1,
  },
  centered: {
    flex: 1,
    justifyContent: 'center',
    alignItems: 'center',
    padding: 16,
  },
  placeholder: {
    padding: 16,
    borderBottomWidth: 1,
    borderBottomColor: 'rgba(0,0,0,0.1)',
  },
})
```

**Why this pattern:**
- Default export for Expo Router (framework requirement)
- Separate screen component (reusability, testing)
- SectionHeader with empty actionLabel + onActionPress (gear button)
- state for modal visibility (local UI state)
- useQuery for feed data (Convex pattern)
- SubscriptionFeedFilters for filtering (reusable component)
- StyleSheet for static layout (performance)
- testID for testing (automation)
