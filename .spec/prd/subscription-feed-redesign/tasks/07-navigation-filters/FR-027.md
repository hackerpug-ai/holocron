EXAMPLES

**Follow this exact modal structure:**

```typescript
// Location: components/subscriptions/SubscriptionSettingsModal.tsx

import { View, ScrollView, StyleSheet, type ViewStyle } from 'react-native'
import { Portal, Dialog, Text, Button, Switch, RadioButton } from 'react-native-paper'
import { useMutation, useQuery } from 'convex/react'
import { api } from '@/convex/_generated/api'

export interface FeedSettings {
  // Notification preferences
  enablePushNotifications: boolean
  enableInAppNotifications: boolean
  // Display options
  showThumbnails: boolean
  autoPlayVideos: boolean
  // Content filter
  contentFilter: 'all' | 'videos-only' | 'blogs-only'
}

interface SubscriptionSettingsModalProps {
  /** Whether modal is visible */
  visible: boolean
  /** Callback to close modal */
  onDismiss: () => void
  /** Navigate to subscriptions management */
  onManageSubscriptions: () => void
  /** Test ID for testing */
  testID?: string
}

export function SubscriptionSettingsModal({
  visible,
  onDismiss,
  onManageSubscriptions,
  testID = 'settings-modal',
}: SubscriptionSettingsModalProps) {
  // Fetch current settings
  const { data: settings } = useQuery(api.users.getFeedSettings, {})

  // Mutation to update settings
  const updateSettings = useMutation(api.users.updateFeedSettings)

  const handleSettingChange = (key: keyof FeedSettings, value: any) => {
    updateSettings({ [key]: value })
  }

  return (
    <Portal>
      <Dialog
        visible={visible}
        onDismiss={onDismiss}
        testID={testID}
      >
        <Dialog.Title>Feed Settings</Dialog.Title>
        <Dialog.ScrollArea>
          <ScrollView style={styles.content}>
            {/* Notification Preferences Section */}
            <Text variant="titleMedium" style={styles.sectionTitle}>
              Notifications
            </Text>

            <View style={styles.settingRow}>
              <View style={styles.settingLabel}>
                <Text variant="bodyMedium">Push notifications</Text>
                <Text variant="bodySmall" style={styles.settingDescription}>
                  Get notified of new content
                </Text>
              </View>
              <Switch
                value={settings?.enablePushNotifications ?? false}
                onValueChange={(value) => handleSettingChange('enablePushNotifications', value)}
                testID={`${testID}-push-switch`}
              />
            </View>

            <View style={styles.settingRow}>
              <View style={styles.settingLabel}>
                <Text variant="bodyMedium">In-app notifications</Text>
                <Text variant="bodySmall" style={styles.settingDescription}>
                  Show badges and banners
                </Text>
              </View>
              <Switch
                value={settings?.enableInAppNotifications ?? false}
                onValueChange={(value) => handleSettingChange('enableInAppNotifications', value)}
                testID={`${testID}-inapp-switch`}
              />
            </View>

            {/* Display Options Section */}
            <Text variant="titleMedium" style={styles.sectionTitle}>
              Display
            </Text>

            <View style={styles.settingRow}>
              <View style={styles.settingLabel}>
                <Text variant="bodyMedium">Show thumbnails</Text>
                <Text variant="bodySmall" style={styles.settingDescription}>
                  Display image thumbnails in feed
                </Text>
              </View>
              <Switch
                value={settings?.showThumbnails ?? true}
                onValueChange={(value) => handleSettingChange('showThumbnails', value)}
                testID={`${testID}-thumbnails-switch`}
              />
            </View>

            <View style={styles.settingRow}>
              <View style={styles.settingLabel}>
                <Text variant="bodyMedium">Auto-play videos</Text>
                <Text variant="bodySmall" style={styles.settingDescription}>
                  Automatically play video previews
                </Text>
              </View>
              <Switch
                value={settings?.autoPlayVideos ?? false}
                onValueChange={(value) => handleSettingChange('autoPlayVideos', value)}
                testID={`${testID}-autoplay-switch`}
              />
            </View>

            {/* Content Filter Section */}
            <Text variant="titleMedium" style={styles.sectionTitle}>
              Content Filter
            </Text>

            <RadioButton.Group
              value={settings?.contentFilter ?? 'all'}
              onValueChange={(value) => handleSettingChange('contentFilter', value)}
              testID={`${testID}-content-filter`}
            >
              <View style={styles.radioRow}>
                <RadioButton value="all" />
                <Text variant="bodyMedium">All content</Text>
              </View>
              <View style={styles.radioRow}>
                <RadioButton value="videos-only" />
                <Text variant="bodyMedium">Videos only</Text>
              </View>
              <View style={styles.radioRow}>
                <RadioButton value="blogs-only" />
                <Text variant="bodyMedium">Blogs only</Text>
              </View>
            </RadioButton.Group>
          </ScrollView>
        </Dialog.ScrollArea>
        <Dialog.Actions>
          <Button onPress={onDismiss} testID={`${testID}-close-button`}>
            Close
          </Button>
          <Button
            onPress={onManageSubscriptions}
            testID={`${testID}-manage-button`}
          >
            Manage Subscriptions
          </Button>
        </Dialog.Actions>
      </Dialog>
    </Portal>
  )
}

const styles = StyleSheet.create({
  content: {
    maxHeight: 400,
  } as ViewStyle,
  sectionTitle: {
    marginTop: 16,
    marginBottom: 8,
  },
  settingRow: {
    flexDirection: 'row',
    justifyContent: 'space-between',
    alignItems: 'center',
    paddingVertical: 8,
  },
  settingLabel: {
    flex: 1,
  },
  settingDescription: {
    opacity: 0.7,
  },
  radioRow: {
    flexDirection: 'row',
    alignItems: 'center',
    paddingVertical: 4,
  },
})
```

**Why this pattern:**
- Portal + Dialog from Paper for modal (accessibility, animation)
- ScrollView in Dialog.ScrollArea for overflow content
- Paper Text, Button, Switch, RadioButton components (consistency)
- Sections grouped with title headers (UX clarity)
- Descriptions under setting labels (UX best practice)
- StyleSheet for static layout (performance)
- testID on all interactive elements (testing)
- Type-safe FeedSettings interface (TypeScript)

**Anti-pattern (DO NOT do this):**

```typescript
// DON'T: Use react-native Text
import { Text } from 'react-native'

// DON'T: Hardcode colors
<Text style={{ color: '#6750A4' }}>Title</Text>

// DON'T: Use inline styles for static layout
<View style={{ flexDirection: 'row', justifyContent: 'space-between' }}>

// DON'T: Skip TypeScript types
interface SubscriptionSettingsModalProps {
  visible: any
  onDismiss: any
}

// DON'T: Create custom toggle/switch
<View onPress={handleToggle}>
  {enabled && <Checkmark />}
</View>
```
