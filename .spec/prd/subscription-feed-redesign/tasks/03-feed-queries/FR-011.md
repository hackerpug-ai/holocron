**Follow this exact pattern (create queries.ts similar to subscriptions):**

```typescript
// Location: convex/feeds/queries.ts

import { query } from "./_generated/server";
import { v } from "convex/values";
import { getFeedArgs } from "./validators";

export const getFeed = query({
  args: {
    limit: v.optional(v.number()),
    contentType: v.optional(v.union(
      v.literal("video"),
      v.literal("blog"),
      v.literal("social")
    )),
    viewed: v.optional(v.boolean()),
    creatorProfileId: v.optional(v.id("creatorProfiles")),
  },
  handler: async (ctx, args) => {
    const limit = args.limit ?? 50;

    // Build index selection based on filters
    if (args.creatorProfileId) {
      // Use by_creator index when filtering by creator
      const items = await ctx.db
        .query("feedItems")
        .withIndex("by_creator", (q) =>
          q.eq("creatorProfileId", args.creatorProfileId)
        )
        .filter((q) => {
          // Apply additional filters
          if (args.contentType && q.field("contentType") !== args.contentType) {
            return false;
          }
          if (args.viewed !== undefined && q.field("viewed") !== args.viewed) {
            return false;
          }
          return true;
        })
        .order("desc")
        .take(limit);

      return items;
    }

    if (args.viewed !== undefined) {
      // Use by_viewed index for unviewed/viewed filtering
      const items = await ctx.db
        .query("feedItems")
        .withIndex("by_viewed", (q) =>
          q.eq("viewed", args.viewed)
        )
        .filter((q) => {
          if (args.contentType && q.field("contentType") !== args.contentType) {
            return false;
          }
          return true;
        })
        .order("desc")
        .take(limit);

      return items;
    }

    // Default: use by_created index
    let query = ctx.db.query("feedItems").withIndex("by_created");

    if (args.contentType) {
      query = query.filter((q) => q.field("contentType") === args.contentType);
    }

    const items = await query.order("desc").take(limit);
    return items;
  },
});
```

**Why this pattern:**
- Index selection based on filter parameters (optimal query performance)
- Filter composition for multiple criteria
- Default limit of 50 items (reasonable for UI)
- Returns items in descending order (newest first)
- Handles all combinations of filters

**Anti-pattern (DO NOT do this):**

```typescript
// DON'T: Use full table scan without indexes
export const getFeed = query({
  args: { ... },
  handler: async (ctx, args) => {
    // BAD: No index usage = slow on large datasets
    const items = await ctx.db.query("feedItems").collect();

    // DON'T: Filter in JavaScript instead of database
    return items.filter(item => {
      if (args.contentType && item.contentType !== args.contentType) {
        return false;
      }
      // More filtering...
    });
  },
});

// DON'T: Return without proper typing
export const getFeed = query({
  handler: async (ctx) => {
    const items = await ctx.db.query("feedItems").take(50);
    return items.map(item => ({
      // Manual transformation = error-prone
      id: item._id,
      title: item.title,
      // Missing fields = incomplete data
    }));
  },
});
```
