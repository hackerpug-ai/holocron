**Target Integration Pattern (useWebView Hook):**

The useWebView hook should be updated to provide a simpler state-based API:

```typescript
// hooks/useWebView.ts (NEW IMPLEMENTATION)
import { useState } from 'react'

export interface WebViewState {
  visible: boolean
  url: string
}

export function useWebView() {
  const [webViewState, setWebViewState] = useState<WebViewState>({
    visible: false,
    url: '',
  })

  const openUrl = (url: string) => {
    setWebViewState({ visible: true, url })
  }

  const closeWebView = () => {
    setWebViewState({ visible: false, url: '' })
  }

  return {
    webViewState,
    openUrl,
    closeWebView,
  }
}

// Usage in feed cards:
import { useWebView } from '@/hooks/useWebView'
import { WebViewSheet } from '@/components/webview/WebViewSheet'

export function VideoFeedCard({ item }) {
  const { webViewState, openUrl, closeWebView } = useWebView()

  return (
    <>
      <Pressable onPress={() => openUrl(item.url)}>
        {/* Card content */}
      </Pressable>

      <WebViewSheet
        visible={webViewState.visible}
        url={webViewState.url}
        onClose={closeWebView}
        testID={`feed-card-${item._id}-webview`}
      />
    </>
  )
}
```

**Feed Card Integration:**

Each feed card variant should use the same pattern:

```typescript
// components/feed/VideoFeedCard.tsx
// components/feed/BlogFeedCard.tsx
// components/feed/SocialFeedCard.tsx

export function VideoFeedCard({ item }) {
  const { webViewState, openUrl, closeWebView } = useWebView()

  const handlePress = () => {
    // Get the best URL from the feed item
    const url = item.primaryLink || item.contentUrl || item.url
    if (url) {
      openUrl(url)
    }
  }

  return (
    <>
      <Pressable onPress={handlePress} testID={`video-feed-card-${item._id}`}>
        {/* Card content */}
      </Pressable>

      <WebViewSheet
        visible={webViewState.visible}
        url={webViewState.url}
        onClose={closeWebView}
        testID={`video-feed-card-${item._id}-webview`}
      />
    </>
  )
}
```

**Anti-patterns (DO NOT do this):**

```typescript
// DON'T: Use Linking.openURL directly
import { Linking } from 'react-native'

const handlePress = () => {
  Linking.openURL(item.url) // ❌ Opens external browser
}

// DON'T: Use router navigation for webview
const router = useRouter()

const handlePress = () => {
  router.push(`/webview/${encodeURIComponent(url)}`) // ❌ Full screen, not sheet
}

// DON'T: Create multiple WebViewSheet instances
return (
  <>
    {items.map(item => (
      <WebViewSheet key={item._id} {...} /> // ❌ Multiple webviews
    ))}
  </>
)
```
