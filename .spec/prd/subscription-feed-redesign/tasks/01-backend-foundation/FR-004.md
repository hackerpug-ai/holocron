EXAMPLES
--------------------------------------------------------------------------------

**Follow this exact pattern (create validators.ts similar to queries/mutations):**

```typescript
// Location: convex/feeds/validators.ts

import { v } from "convex/values";

// Content type enum
export const contentTypeEnum = v.union(
  v.literal("video"),
  v.literal("blog"),
  v.literal("social"),
  v.literal("mixed")
);

// Feed item validators
export const feedItemFields = {
  groupKey: v.string(),
  title: v.string(),
  summary: v.optional(v.string()),
  contentType: contentTypeEnum,
  itemCount: v.number(),
  itemIds: v.array(v.string()),
  creatorProfileId: v.optional(v.id("creatorProfiles")),
  subscriptionIds: v.array(v.id("subscriptionSources")),
  thumbnailUrl: v.optional(v.string()),
  viewed: v.boolean(),
  viewedAt: v.optional(v.number()),
  publishedAt: v.number(),
  discoveredAt: v.number(),
  createdAt: v.number(),
};

// Feed session validators
export const feedSessionFields = {
  startTime: v.number(),
  endTime: v.optional(v.number()),
  itemsViewed: v.number(),
  itemsConsumed: v.number(),
  sessionSource: v.optional(v.string()),
};

// Query argument validators
export const getFeedArgs = {
  limit: v.optional(v.number()),
  contentType: v.optional(contentTypeEnum),
  viewed: v.optional(v.boolean()),
  creatorProfileId: v.optional(v.id("creatorProfiles")),
};

// Mutation argument validators
export const markViewedArgs = {
  feedItemIds: v.array(v.id("feedItems")),
  viewedAt: v.optional(v.number()),
};

export const markAllViewedArgs = {
  viewedAt: v.optional(v.number()),
};
```

**Why this pattern:**
- Export individual field objects for reusability
- Export argument validators for queries/mutations
- Define enums once, reuse across validators
- Use v.optional() consistently with schema

**Anti-pattern (DO NOT do this):**

```typescript
// DON'T: Inline validators instead of exporting
export const getFeedArgs = {
  contentType: v.optional(v.union(
    v.literal("video"),
    v.literal("blog"),
    v.literal("social"),
    v.literal("mixed")
  )),
};

// DON'T: Create validators that don't match schema
export const feedItemFields = {
  groupKey: v.string(),
  title: v.string(),
  // Missing fields = schema mismatch
};
```
