**Follow this exact pattern:**

```typescript
// Location: convex/feeds/mutations.ts (add to existing file)

import { createDigestNotificationArgs } from "./validators";
import { api } from "./_generated/api";

export const createDigestNotification = mutation({
  args: createDigestNotificationArgs,
  handler: async (ctx, args) => {
    // Get current user (this will be called from cron or context)
    const identity = await ctx.auth.getUserIdentity();
    if (!identity) {
      throw new Error("Unauthorized: No user identity found");
    }

    // Get digest summary using internal query
    const summary = await ctx.runQuery(
      api.feeds.getDigestSummary,
      {} // No args needed for summary
    );

    // Don't create notification if no unviewed items
    if (summary.totalItems === 0) {
      return { created: false, reason: "No unviewed items" };
    }

    // Create notification
    const notificationId = await ctx.db.insert("notifications", {
      userId: identity.subject,
      type: "feed_digest",
      title: `You have ${summary.totalItems} new items to explore`,
      message: buildDigestMessage(summary),
      metadata: {
        totalItems: summary.totalItems,
        breakdown: summary.byContentType,
        sampleItems: summary.sampleItems.slice(0, 5),
      },
      read: false,
      createdAt: Date.now(),
    });

    return { created: true, notificationId };
  },
});

// Helper function to build digest message
function buildDigestMessage(summary: any): string {
  const { byContentType, totalItems } = summary;
  const parts: string[] = [];

  if (byContentType.video > 0) {
    parts.push(`${byContentType.video} videos`);
  }
  if (byContentType.blog > 0) {
    parts.push(`${byContentType.blog} articles`);
  }
  if (byContentType.social > 0) {
    parts.push(`${byContentType.social} social posts`);
  }

  const itemList = parts.join(", ");
  return `${totalItems} new feed items: ${itemList}`;
}
```

**Why this pattern:**
- Check auth for user identity
- Run getDigestSummary query to get data
- Early return if no unviewed items (no spam)
- Use feed_digest type for notifications
- Include rich metadata for UI rendering
- Helper function for message formatting
