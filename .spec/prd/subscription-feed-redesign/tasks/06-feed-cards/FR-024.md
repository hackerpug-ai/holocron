Example blog card structure:

```typescript
// Inside SubscriptionFeedItem.tsx

import { View, StyleSheet } from 'react-native'
import { Text } from '@/components/ui/text'
import { User, Clock, Calendar } from '@/components/ui/icons'
import { cn } from '@/lib/utils'

interface BlogFeedCardProps {
  title: string
  summary?: string
  creatorName?: string
  wordCount?: number
  publishedAt: number
}

function calculateReadTime(wordCount: number): number {
  // Average reading speed: 200 words per minute
  return Math.max(1, Math.ceil(wordCount / 200))
}

export function BlogFeedCard({
  title,
  summary,
  creatorName,
  wordCount,
  publishedAt,
}: BlogFeedCardProps) {
  const { semantic } = useSemanticTheme()
  const readTime = wordCount ? calculateReadTime(wordCount) : null

  return (
    <View style={[styles.blogContainer, { gap: semantic.space.md }]}>
      {/* Title */}
      <Text
        variant="titleLarge"
        numberOfLines={2}
        style={styles.title}
        testID="feed-item-blog-title"
      >
        {title}
      </Text>

      {/* Excerpt */}
      {summary && (
        <Text
          variant="bodyMedium"
          numberOfLines={3}
          className="text-muted-foreground"
          style={styles.excerpt}
          testID="feed-item-blog-excerpt"
        >
          {summary}
        </Text>
      )}

      {/* Metadata row */}
      <View style={[styles.metadata, { gap: semantic.space.md }]}>
        {/* Author */}
        {creatorName && (
          <View style={styles.metaItem}>
            <User size={14} className="text-muted-foreground" />
            <Text variant="labelSmall" className="text-muted-foreground">
              {creatorName}
            </Text>
          </View>
        )}

        {/* Read time */}
        {readTime && (
          <View style={styles.metaItem}>
            <Clock size={14} className="text-muted-foreground" />
            <Text variant="labelSmall" className="text-muted-foreground">
              {readTime} min read
            </Text>
          </View>
        )}

        {/* Published date */}
        <View style={styles.metaItem}>
          <Calendar size={14} className="text-muted-foreground" />
          <Text variant="labelSmall" className="text-muted-foreground">
            {formatDate(publishedAt)}
          </Text>
        </View>
      </View>
    </View>
  )
}

const styles = StyleSheet.create({
  blogContainer: {
    width: '100%',
    paddingVertical: 4,
  },
  title: {
    fontWeight: '600',
  },
  excerpt: {
    lineHeight: 22,
  },
  metadata: {
    flexDirection: 'row',
    flexWrap: 'wrap',
  },
  metaItem: {
    flexDirection: 'row',
    gap: 6,
    alignItems: 'center',
  },
})
```
