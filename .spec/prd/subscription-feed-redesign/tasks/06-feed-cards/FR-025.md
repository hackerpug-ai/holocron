Example social card structure:

```typescript
// Inside SubscriptionFeedItem.tsx

import { View, StyleSheet } from 'react-native'
import { Text } from '@/components/ui/text'
import { Avatar, AvatarFallback, AvatarImage } from '@/components/ui/avatar'
import { PlatformBadge } from './PlatformBadge'
import { Heart, MessageCircle, Share2, Clock } from '@/components/ui/icons'
import { cn } from '@/lib/utils'

interface SocialFeedCardProps {
  authorName: string
  authorHandle: string
  authorAvatarUrl?: string
  content?: string
  platform: 'twitter' | 'bluesky' | 'github' | 'reddit'
  likesCount?: number
  commentsCount?: number
  sharesCount?: number
  publishedAt: number
}

function formatEngagement(count: number): string {
  if (count >= 1000000) return `${(count / 1000000).toFixed(1)}M`
  if (count >= 1000) return `${(count / 1000).toFixed(1)}K`
  return count.toString()
}

function formatRelativeTime(timestamp: number): string {
  const now = Date.now()
  const diff = now - timestamp
  const minutes = Math.floor(diff / 60000)
  const hours = Math.floor(diff / 3600000)
  const days = Math.floor(diff / 86400000)

  if (minutes < 60) return `${minutes}m ago`
  if (hours < 24) return `${hours}h ago`
  return `${days}d ago`
}

function getInitials(name: string): string {
  return name
    .split(' ')
    .map((word) => word[0])
    .join('')
    .toUpperCase()
    .slice(0, 2)
}

export function SocialFeedCard({
  authorName,
  authorHandle,
  authorAvatarUrl,
  content,
  platform,
  likesCount,
  commentsCount,
  sharesCount,
  publishedAt,
}: SocialFeedCardProps) {
  const { semantic } = useSemanticTheme()

  return (
    <View style={[styles.socialContainer, { gap: semantic.space.md }]}>
      {/* Author row */}
      <View style={[styles.authorRow, { gap: semantic.space.sm }]}>
        {/* Avatar */}
        <Avatar className="h-10 w-10" alt={authorName}>
          {authorAvatarUrl ? (
            <AvatarImage source={{ uri: authorAvatarUrl }} />
          ) : (
            <AvatarFallback>
              <Text variant="labelSmall" className="text-foreground font-semibold">
                {getInitials(authorName)}
              </Text>
            </AvatarFallback>
          )}
        </Avatar>

        {/* Author info */}
        <View style={styles.authorInfo}>
          <View style={[styles.authorNameRow, { gap: semantic.space.xs }]}>
            <Text variant="bodyMedium" className="font-semibold">
              {authorName}
            </Text>
            <PlatformBadge
              platform={platform}
              handle={authorHandle}
              selected={false}
              onPress={undefined}
              className="pressable-opacity"
            />
          </View>
          <Text variant="labelSmall" className="text-muted-foreground">
            @{authorHandle} • {formatRelativeTime(publishedAt)}
          </Text>
        </View>
      </View>

      {/* Content preview */}
      {content && (
        <Text
          variant="bodyMedium"
          numberOfLines={4}
          className="text-foreground"
          style={styles.content}
          testID="feed-item-social-content"
        >
          {content}
        </Text>
      )}

      {/* Engagement stats */}
      <View style={[styles.engagement, { gap: semantic.space.md }]}>
        {likesCount !== undefined && (
          <View style={styles.statItem}>
            <Heart size={14} className="text-muted-foreground" />
            <Text variant="labelSmall" className="text-muted-foreground">
              {formatEngagement(likesCount)}
            </Text>
          </View>
        )}

        {commentsCount !== undefined && (
          <View style={styles.statItem}>
            <MessageCircle size={14} className="text-muted-foreground" />
            <Text variant="labelSmall" className="text-muted-foreground">
              {formatEngagement(commentsCount)}
            </Text>
          </View>
        )}

        {sharesCount !== undefined && (
          <View style={styles.statItem}>
            <Share2 size={14} className="text-muted-foreground" />
            <Text variant="labelSmall" className="text-muted-foreground">
              {formatEngagement(sharesCount)}
            </Text>
          </View>
        )}
      </View>
    </View>
  )
}

const styles = StyleSheet.create({
  socialContainer: {
    width: '100%',
    paddingVertical: 4,
  },
  authorRow: {
    flexDirection: 'row',
    alignItems: 'flex-start',
  },
  authorInfo: {
    flex: 1,
    gap: 2,
  },
  authorNameRow: {
    flexDirection: 'row',
    alignItems: 'center',
  },
  content: {
    lineHeight: 22,
  },
  engagement: {
    flexDirection: 'row',
  },
  statItem: {
    flexDirection: 'row',
    gap: 6,
    alignItems: 'center',
  },
})
```
