Example component structure from CreatorGroupCard:

```typescript
import { View, Pressable, StyleSheet } from 'react-native'
import { Text } from '@/components/ui/text'
import { useSemanticTheme } from '@/hooks/use-semantic-theme'
import Animated, { useAnimatedStyle, withSpring } from 'react-native-reanimated'

interface FeedItemProps {
  feedItemId: string
  groupKey: string
  title: string
  summary?: string
  contentType: 'video' | 'blog' | 'social'
  itemCount: number
  thumbnailUrl?: string
  viewed: boolean
  publishedAt: number
  onPress?: () => void
}

const AnimatedPressable = Animated.createAnimatedComponent(Pressable)

export function SubscriptionFeedItem({
  feedItemId,
  groupKey,
  title,
  summary,
  contentType,
  itemCount,
  thumbnailUrl,
  viewed,
  publishedAt,
  onPress,
}: FeedItemProps) {
  const { semantic } = useSemanticTheme()
  const [pressed, setPressed] = useState(false)

  const animatedStyle = useAnimatedStyle(() => {
    return {
      transform: [{ scale: withSpring(pressed ? 0.98 : 1) }]
    }
  })

  return (
    <AnimatedPressable
      onPress={onPress}
      onPressIn={() => setPressed(true)}
      onPressOut={() => setPressed(false)}
      style={[
        styles.container,
        animatedStyle,
        {
          opacity: viewed ? 0.6 : 1,
          backgroundColor: semantic.color.card.default,
          borderColor: semantic.color.border.default,
          borderRadius: semantic.radius.lg,
          padding: semantic.space.lg,
        }
      ]}
      testID={`feed-item-${feedItemId}`}
    >
      {/* Variant content rendered here */}
    </AnimatedPressable>
  )
}

const styles = StyleSheet.create({
  container: {
    borderWidth: 1,
    width: '100%',
  },
})
```
