```typescript
// Location: convex/feeds/internal.ts

import { v } from "convex/values";
import { internalAction, internalMutation } from "../_generated/server";
import { internal } from "../_generated/api";

// ============================================================================
// Feed Building Action
// ============================================================================

/**
 * Build feed from recent subscription content
 *
 * Groups content by creatorProfileId (or identifier) and creates feed items.
 * This action is called by cron job every 2 hours.
 *
 * Idempotent: Safe to run multiple times - skips content already in feed.
 */
export const buildFeed = internalAction({
  args: {},
  handler: async (ctx) => {
    const now = Date.now();

    // Step 1: Query recent unprocessed content (last 7 days, not in feed)
    const recentContent = await ctx.runQuery(internal.feeds.internal.getUnprocessedContent, {
      since: now - 7 * 24 * 60 * 60 * 1000, // 7 days ago
    });

    // Step 2: Group by creator
    const groupedByCreator = new Map<string, typeof recentContent>();

    for (const content of recentContent) {
      const groupKey = content.creatorProfileId
        ? `profile:${content.creatorProfileId}`
        : `identifier:${content.identifier}`;

      if (!groupedByCreator.has(groupKey)) {
        groupedByCreator.set(groupKey, []);
      }
      groupedByCreator.get(groupKey)!.push(content);
    }

    // Step 3: Create feed items for each group
    const results = [];
    for (const [groupKey, contents] of groupedByCreator.entries()) {
      const firstContent = contents[0];

      // Determine content type (majority wins)
      const typeCounts = { video: 0, blog: 0, social: 0 };
      contents.forEach(c => {
        if (c.contentType) typeCounts[c.contentType as keyof typeof typeCounts]++;
      });
      const dominantType = (Object.entries(typeCounts)
        .sort((a, b) => b[1] - a[1])[0][0]) as "video" | "blog" | "social";

      // Create feed item
      const feedItemId = await ctx.runMutation(internal.feeds.internal.createFeedItem, {
        groupKey,
        title: firstContent.creatorName || firstContent.identifier,
        summary: contents.length === 1
          ? firstContent.title
          : `${contents.length} new items from ${firstContent.creatorName || firstContent.identifier}`,
        contentType: dominantType,
        itemCount: contents.length,
        itemIds: contents.map(c => c._id),
        creatorProfileId: firstContent.creatorProfileId,
        subscriptionIds: [...new Set(contents.map(c => c.subscriptionId))],
        thumbnailUrl: contents.find(c => c.thumbnailUrl)?.thumbnailUrl,
        publishedAt: Math.max(...contents.map(c => c.publishedAt || 0)),
        discoveredAt: Math.min(...contents.map(c => c.discoveredAt || now)),
        createdAt: now,
      });

      // Mark content as in feed
      await ctx.runMutation(internal.feeds.internal.markContentInFeed, {
        contentIds: contents.map(c => c._id),
        feedItemId,
      });

      results.push({ groupKey, feedItemId, itemCount: contents.length });
    }

    return {
      processed: results.length,
      items: results,
    };
  },
});

// ============================================================================
// Internal Queries (Helper Functions)
// ============================================================================

export const getUnprocessedContent = internalQuery({
  args: {
    since: v.number(),
  },
  handler: async (ctx, args) => {
    const content = await ctx.db
      .query("subscriptionContent")
      .withIndex("by_discoveredAt", (q) =>
        q.gte("discoveredAt", args.since)
      )
      .filter((q) => q.eq(q.field("inFeed"), false))
      .collect();

    return content;
  },
});

// ============================================================================
// Internal Mutations (Helper Functions)
// ============================================================================

export const createFeedItem = internalMutation({
  args: {
    groupKey: v.string(),
    title: v.string(),
    summary: v.optional(v.string()),
    contentType: v.union(
      v.literal("video"),
      v.literal("blog"),
      v.literal("social"),
      v.literal("mixed")
    ),
    itemCount: v.number(),
    itemIds: v.array(v.id("subscriptionContent")),
    creatorProfileId: v.optional(v.id("creatorProfiles")),
    subscriptionIds: v.array(v.id("subscriptionSources")),
    thumbnailUrl: v.optional(v.string()),
    publishedAt: v.number(),
    discoveredAt: v.number(),
    createdAt: v.number(),
  },
  handler: async (ctx, args) => {
    const feedItemId = await ctx.db.insert("feedItems", {
      groupKey: args.groupKey,
      title: args.title,
      summary: args.summary,
      contentType: args.contentType,
      itemCount: args.itemCount,
      itemIds: args.itemIds,
      creatorProfileId: args.creatorProfileId,
      subscriptionIds: args.subscriptionIds,
      thumbnailUrl: args.thumbnailUrl,
      viewed: false,
      viewedAt: undefined,
      publishedAt: args.publishedAt,
      discoveredAt: args.discoveredAt,
      createdAt: args.createdAt,
    });

    return feedItemId;
  },
});

export const markContentInFeed = internalMutation({
  args: {
    contentIds: v.array(v.id("subscriptionContent")),
    feedItemId: v.id("feedItems"),
  },
  handler: async (ctx, args) => {
    for (const contentId of args.contentIds) {
      const content = await ctx.db.get(contentId);
      if (content) {
        await ctx.db.patch(contentId, {
          feedItemId: args.feedItemId,
          inFeed: true,
        });
      }
    }
  },
});
```

**Why this pattern:**
- Internal action cannot be called from client (security)
- Separates query/mutation helpers for testability
- Groups by creatorProfileId OR identifier (fallback)
- Uses Map for efficient grouping
- Determines contentType by majority vote
- Marks content with inFeed flag for idempotency
- Returns results for logging/monitoring

```typescript
// DON'T: Create duplicates on re-run
export const buildFeed = internalAction({
  args: {},

    // Missing inFeed filter = duplicates every time!

    // ...
  },
});

// DON'T: Don't handle missing creatorProfileId
const groupKey = content.creatorProfileId; // Crashes if undefined!

// DON'T: Use client-facing mutation
export const buildFeed = mutation({  // Wrong - use internalAction
  args: {},
  handler: async (ctx, args) => {
    // ...
  },
});
```
