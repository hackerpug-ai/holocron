```ts
export interface Fixture {
  id: string;
  query: string;
  expectedIntent: string;
  expectedQueryShape: string;
  expectedTool: string | null;
  mockLlmResponse: string; // raw JSON string the mock returns
}

export const FIXTURES: Fixture[] = [
  {
    id: 'canonical-failing',
    query: 'career coaches for people with autism in San Francisco — provide 3-5 highly rated referrals',
    expectedIntent: 'research',
    expectedQueryShape: 'recommendation',
    expectedTool: 'find_recommendations',
    mockLlmResponse: '{"intent":"research","queryShape":"recommendation","confidence":"high","reasoning":"User wants named SF providers"}',
  },
  // ...29 more covering recommendation(8), factual(6), comprehensive(4), exploratory(3), ambiguous(5), conversation(4), knowledge(3)
];

// runEval.test.ts
import { describe, it, expect, vi } from 'vitest';
import { FIXTURES } from './fixtures';
import { classifyIntent } from '../triage';

describe('routing eval harness', () => {
  it('canonical failing case', async () => {
    const f = FIXTURES.find(x => x.id === 'canonical-failing')!;
    const r = await runClassifyMocked(f);
    expect(r.intent).toBe('research');
    expect(r.queryShape).toBe('recommendation');
    expect(r.confidence).toBe('high');
  });

  it('fixture count >= 30', () => {
    expect(FIXTURES.length).toBeGreaterThanOrEqual(30);
  });

  it('accuracy gate', async () => {
    let pass = 0;
    for (const f of FIXTURES) {
      const r = await runClassifyMocked(f);
      if (r.intent === f.expectedIntent && r.queryShape === f.expectedQueryShape) pass++;
    }
    const accuracy = pass / FIXTURES.length;
    expect(accuracy).toBeGreaterThanOrEqual(0.9);
  });
});
```
