```ts
import { internalAction, action } from '../_generated/server';
import { v } from 'convex/values';
import { RecommendationSynthesisSchema, RECOMMENDATION_SYNTHESIS_PROMPT } from '../chat/specialistPrompts';

const recArgs = {
  query: v.string(),
  count: v.optional(v.number()),
  location: v.optional(v.string()),
  constraints: v.optional(v.array(v.string())),
};

const recReturn = v.object({
  items: v.array(v.object({
    name: v.string(),
    description: v.string(),
    whyRecommended: v.string(),
    rating: v.optional(v.number()),
    location: v.optional(v.string()),
    pricing: v.optional(v.string()),
    contact: v.optional(v.object({
      phone: v.optional(v.string()),
      url: v.optional(v.string()),
      email: v.optional(v.string()),
    })),
  })),
  sources: v.array(v.object({
    title: v.string(),
    url: v.string(),
    snippet: v.string(),
  })),
  query: v.string(),
  durationMs: v.number(),
});

async function findRecommendationsCore(args: CoreArgs): Promise<RecommendationResult> {
  const start = Date.now();
  const controller = new AbortController();
  const timer = setTimeout(() => controller.abort(), 30_000);
  try {
    const enhanced = buildEnhancedQuery(args);
    const sources = await jinaSearch(enhanced, { signal: controller.signal });
    if (sources.length === 0) {
      return { items: [], sources: [], query: args.query, durationMs: Date.now() - start };
    }
    const content = await parallelJinaReader(sources.slice(0, 5), { signal: controller.signal, timeoutMs: 15_000 });
    let synth = await synthesize(content, args, { signal: controller.signal });
    let parsed = RecommendationSynthesisSchema.safeParse(synth);
    if (!parsed.success) {
      synth = await synthesize(content, args, { signal: controller.signal, strict: true });
      parsed = RecommendationSynthesisSchema.safeParse(synth);
    }
    if (!parsed.success) {
      return { items: [], sources, query: args.query, durationMs: Date.now() - start };
    }
    return {
      items: parsed.data.items,
      sources: parsed.data.sources,
      query: args.query,
      durationMs: Date.now() - start,
    };
  } finally {
    clearTimeout(timer);
  }
}

export const findRecommendationsAction = internalAction({
  args: recArgs,
  returns: recReturn,
  handler: (_ctx, args) => findRecommendationsCore(args),
});

// PUBLIC — exposed to MCP (REC-MCP-01)
export const findRecommendations = action({
  args: recArgs,
  returns: recReturn,
  handler: (_ctx, args) => findRecommendationsCore(args),
});
```
