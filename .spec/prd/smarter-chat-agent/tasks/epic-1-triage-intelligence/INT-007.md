CODE PATTERN (Reference)

**Source**: `convex/chat` existing queries using withIndex

```ts
import { query } from '../_generated/server';
import { v } from 'convex/values';

export const listRecentClassifications = query({
  args: { limit: v.optional(v.number()) },
  returns: v.array(v.object({
    _id: v.id('agentTelemetry'),
    intent: v.string(),
    queryShape: v.string(),
    classificationSource: v.string(),
    createdAt: v.number(),
  })),
  handler: async (ctx, { limit = 20 }) => {
    return await ctx.db
      .query('agentTelemetry')
      .withIndex('by_createdAt')
      .order('desc')
      .take(limit);
  },
});

export const getCurrentPhase = query({
  args: { conversationId: v.id('conversations') },
  returns: v.object({
    phase: v.string(),
    toolName: v.union(v.string(), v.null()),
  }),
  handler: async (ctx, { conversationId }) => {
    const recent = await ctx.db
      .query('agentTelemetry')
      .withIndex('by_conversation', q => q.eq('conversationId', conversationId))
      .order('desc')
      .first();
    if (!recent) return { phase: 'idle', toolName: null };
    const activeToolsCalled = recent.toolsCalled?.[0];
    // phase mapping rules: if toolsCalled set and very recent, phase='tool_execution';
    // else if rawLlmResponse set but no toolsCalled, phase='synthesis'; else 'idle'
    const age = Date.now() - recent.createdAt;
    if (age < 10_000 && activeToolsCalled) {
      return { phase: 'tool_execution', toolName: activeToolsCalled };
    }
    return { phase: 'idle', toolName: null };
  },
});
```
